  // ============================================================
  // DATA EXTRACTION - Pulls metric data from RTB page DOM
  // ============================================================
  class RTBDataExtractor {
    constructor() {
      this.metrics = [];
      this.comments = [];
      this.historicalData = [];
      this.lastUpdate = null;
    }

    extractAll() {
      this.metrics = [];
      this.comments = [];
      this.extractMetrics();
      this.extractComments();
      this.extractHistoricalData();
      this.lastUpdate = new Date();
      console.log('[RTB Analyzer] Extracted', this.metrics.length, 'metrics,', this.comments.length, 'comments');
      return { metrics: this.metrics, comments: this.comments, historical: this.historicalData };
    }

    extractMetrics() {
      // Parse the RTB Action Plans table from the currently active AWSUI tab
      console.log('[RTB Analyzer] Starting extraction...');

      // Find the active AWSUI tab panel
      let panel = null;

      // Strategy A: button with aria-selected="true" and aria-controls
      const activeTab = document.querySelector('button[aria-selected="true"][aria-controls]');
      if (activeTab) {
        const panelId = activeTab.getAttribute('aria-controls');
        console.log('[RTB Analyzer] Active tab:', activeTab.textContent.trim(), '| Panel:', panelId);
        panel = document.getElementById(panelId);
      }

      // Strategy B: any element with role="tab" and aria-selected="true"
      if (!panel) {
        const roleTab = document.querySelector('[role="tab"][aria-selected="true"]');
        if (roleTab) {
          const panelId = roleTab.getAttribute('aria-controls');
          console.log('[RTB Analyzer] Found role=tab:', roleTab.textContent.trim(), '| Panel:', panelId);
          if (panelId) panel = document.getElementById(panelId);
        }
      }

      // Strategy C: look for visible tabpanel
      if (!panel) {
        const tabPanels = document.querySelectorAll('[role="tabpanel"]');
        for (const tp of tabPanels) {
          if (tp.offsetParent !== null && tp.textContent.trim().length > 50) {
            console.log('[RTB Analyzer] Using visible tabpanel');
            panel = tp;
            break;
          }
        }
      }

      // Strategy D: skip tab detection entirely, search the whole page
      if (!panel) {
        console.log('[RTB Analyzer] No active tab/panel found — searching entire page');
        panel = document.body;
      }

      // Debug: what's inside the panel?
      const tables = panel.querySelectorAll('table');
      const grids = panel.querySelectorAll('[role="table"], [role="grid"]');
      const rows = panel.querySelectorAll('[role="row"]');
      const trs = panel.querySelectorAll('tr');
      console.log('[RTB Analyzer] Panel contents - tables:', tables.length, 'grids:', grids.length, 'role=row:', rows.length, 'tr:', trs.length);

      // Strategy 1: Standard <table> element — try all tables
      if (tables.length > 0) {
        console.log('[RTB Analyzer] Trying', tables.length, '<table> elements');
        for (let t = 0; t < tables.length && this.metrics.length === 0; t++) {
          console.log('[RTB Analyzer] Trying table', t, '- rows:', tables[t].querySelectorAll('tr').length);
          this.parseRTBActionPlanTable(tables[t]);
        }
      }

      // Strategy 2: AWSUI uses role="table" or role="grid" — try all grids
      if (this.metrics.length === 0 && grids.length > 0) {
        console.log('[RTB Analyzer] Trying', grids.length, 'role=table/grid elements');
        for (let g = 0; g < grids.length && this.metrics.length === 0; g++) {
          this.parseAWSUITable(grids[g]);
        }
      }

      // Strategy 3: role="row" elements directly in panel
      if (this.metrics.length === 0 && rows.length > 0) {
        console.log('[RTB Analyzer] Using role=row elements');
        this.parseAWSUIRows(rows);
      }

      // Strategy 4: <tr> elements (table rows without proper table wrapper)
      if (this.metrics.length === 0 && trs.length > 0) {
        console.log('[RTB Analyzer] Using <tr> elements');
        this.parseRTBFromTRs(trs);
      }

      // Strategy 5: Last resort - look for the table anywhere on page using AWSUI table class patterns
      if (this.metrics.length === 0) {
        console.log('[RTB Analyzer] Trying AWSUI class-based table detection');
        const awsuiTable = panel.querySelector('[class*="table"], [class*="grid"], [class*="collection"]');
        if (awsuiTable) {
          console.log('[RTB Analyzer] Found AWSUI table-like element, class:', awsuiTable.className.substring(0, 100));
          // Try to find header and body within it
          this.parseAWSUIGeneric(awsuiTable);
        }
      }

      // Deduplicate by metric name + path
      const seen = new Set();
      this.metrics = this.metrics.filter(m => {
        const key = `${m.name}-${m.path || ''}-${m.actionPlanId || ''}`;
        if (seen.has(key)) return false;
        seen.add(key);
        return true;
      });

      console.log('[RTB Analyzer] Final metric count:', this.metrics.length);
    }

    parseAWSUITable(grid) {
      // AWSUI table component uses role="table" with role="row" children
      const allRows = grid.querySelectorAll('[role="row"]');
      if (allRows.length < 2) return;
      this.parseAWSUIRows(allRows);
    }

    parseAWSUIRows(allRows) {
      // First row with role="columnheader" cells is the header
      let headerRow = null;
      let dataRows = [];

      for (const row of allRows) {
        const hasHeaders = row.querySelector('[role="columnheader"], th');
        if (hasHeaders && !headerRow) {
          headerRow = row;
        } else {
          dataRows.push(row);
        }
      }

      if (!headerRow) {
        // Maybe first row is header
        headerRow = allRows[0];
        dataRows = Array.from(allRows).slice(1);
      }

      const headers = Array.from(headerRow.querySelectorAll('[role="columnheader"], th, [class*="header-cell"]'))
        .map(h => h.textContent.replace(/[▲▼↑↓⬆⬇▾▴]/g, '').trim().toLowerCase());

      console.log('[RTB Analyzer] AWSUI headers:', headers.join(' | '));
      console.log('[RTB Analyzer] AWSUI data rows:', dataRows.length);

      if (headers.length === 0) return;

      const colMap = this.buildColumnMap(headers);
      console.log('[RTB Analyzer] Column map:', JSON.stringify(colMap));

      dataRows.forEach((row, idx) => {
        const cells = row.querySelectorAll('[role="gridcell"], [role="cell"], td, [class*="body-cell"]');
        if (cells.length < 3) return;
        this.parseRTBDataRow(cells, colMap, row, idx);
      });
    }

    parseAWSUIGeneric(container) {
      // Try to find header cells and body cells by class patterns
      const headerCells = container.querySelectorAll('[class*="header-cell"], [class*="headerCell"], thead th');
      const bodyCells = container.querySelectorAll('[class*="body-cell"], [class*="bodyCell"]');

      if (headerCells.length === 0) return;

      const headers = Array.from(headerCells)
        .map(h => h.textContent.replace(/[▲▼↑↓⬆⬇▾▴]/g, '').trim().toLowerCase());
      console.log('[RTB Analyzer] Generic headers:', headers.join(' | '));

      if (bodyCells.length === 0) return;

      // Group body cells into rows based on header count
      const colCount = headers.length;
      const colMap = this.buildColumnMap(headers);
      const rowCount = Math.floor(bodyCells.length / colCount);
      console.log('[RTB Analyzer] Generic: cols:', colCount, 'body cells:', bodyCells.length, 'rows:', rowCount);

      for (let i = 0; i < rowCount; i++) {
        const rowCells = Array.from(bodyCells).slice(i * colCount, (i + 1) * colCount);
        this.parseRTBDataRow(rowCells, colMap, null, i);
      }
    }

    parseRTBFromTRs(trs) {
      if (trs.length < 2) return;
      const headerRow = trs[0];
      const headers = Array.from(headerRow.querySelectorAll('th, td'))
        .map(h => h.textContent.replace(/[▲▼↑↓⬆⬇▾▴]/g, '').trim().toLowerCase());

      if (!headers.some(h => h.includes('metric'))) return;

      const colMap = this.buildColumnMap(headers);
      for (let i = 1; i < trs.length; i++) {
        const cells = trs[i].querySelectorAll('td');
        if (cells.length < 3) continue;
        this.parseRTBDataRow(cells, colMap, trs[i], i - 1);
      }
    }

    buildColumnMap(headers) {
      return {
        actionPlanId: headers.findIndex(h => h.includes('action plan') || h.includes('plan id')),
        path: headers.findIndex(h => h === 'path'),
        metric: headers.findIndex(h => h === 'metric' || (h.includes('metric') && !h.includes('glidepath'))),
        status: headers.findIndex(h => h === 'status'),
        category: headers.findIndex(h => h === 'category'),
        program: headers.findIndex(h => h === 'program'),
        initialValue: headers.findIndex(h => h.includes('initial') || (h.includes('value') && !h.includes('target'))),
        target: headers.findIndex(h => h === 'target'),
        glidePathStatus: headers.findIndex(h => h.includes('glide') && h.includes('status')),
        pctToGlidepath: headers.findIndex(h => h.includes('% to') && h.includes('glidepath')),
        pctToTarget: headers.findIndex(h => h.includes('% to') && h.includes('target')),
        siteOwner: headers.findIndex(h => h.includes('owner')),
        startDate: headers.findIndex(h => h.includes('start date') || (h.includes('date') && !h.includes('update'))),
      };
    }

    parseRTBDataRow(cells, colMap, rowElement, idx) {
      const cellTexts = Array.from(cells).map(c => c.textContent.trim());
      if (idx === 0) console.log('[RTB Analyzer] First data row:', cellTexts.join(' | '));

      const actionPlanId = colMap.actionPlanId >= 0 ? cellTexts[colMap.actionPlanId] : '';
      const path = colMap.path >= 0 ? cellTexts[colMap.path] : '';
      const metricName = colMap.metric >= 0 ? cellTexts[colMap.metric] : '';
      const status = colMap.status >= 0 ? cellTexts[colMap.status] : '';
      const category = colMap.category >= 0 ? cellTexts[colMap.category] : '';
      const program = colMap.program >= 0 ? cellTexts[colMap.program] : '';
      const initialValueText = colMap.initialValue >= 0 ? cellTexts[colMap.initialValue] : '';
      const targetText = colMap.target >= 0 ? cellTexts[colMap.target] : '';
      const glidePathStatus = colMap.glidePathStatus >= 0 ? cellTexts[colMap.glidePathStatus] : '';
      const pctToGlidepathText = colMap.pctToGlidepath >= 0 ? cellTexts[colMap.pctToGlidepath] : '';
      const pctToTargetText = colMap.pctToTarget >= 0 ? cellTexts[colMap.pctToTarget] : '';
      const siteOwner = colMap.siteOwner >= 0 ? cellTexts[colMap.siteOwner] : '';
      const startDate = colMap.startDate >= 0 ? cellTexts[colMap.startDate] : '';

      if (!metricName && !actionPlanId) return;

      const initialValue = this.parseNumericValue(initialValueText);
      const target = this.parseNumericValue(targetText);
      const pctToTarget = this.parseRTBPercentage(pctToTargetText);
      const pctToGlidepath = this.parseRTBPercentage(pctToGlidepathText);
      const metricStatus = this.parseRTBStatus(glidePathStatus, pctToTarget);

      let percentOfTarget = null;
      if (initialValue !== null && target !== null && target !== 0) {
        percentOfTarget = ((initialValue / target) * 100).toFixed(1);
      }

      this.metrics.push({
        name: metricName,
        actionPlanId, path: path || '—',
        value: initialValue, target,
        status: metricStatus,
        category: category || 'Other',
        program: program || '',
        glidePathStatus, pctToGlidepath, pctToTarget,
        percentOfTarget, siteOwner, startDate,
        planStatus: status,
        element: rowElement,
        rawText: (rowElement || cells[0]?.parentElement)?.textContent?.trim().substring(0, 300) || '',
      });
    }

    findAWSUIActiveTable() { return null; }

    isElementVisible(el) {
      if (!el) return false;
      let current = el;
      while (current && current !== document.body) {
        const style = window.getComputedStyle(current);
        if (style.display === 'none' || style.visibility === 'hidden') return false;
        if (current.getAttribute('aria-hidden') === 'true') return false;
        if (current.hasAttribute('hidden')) return false;
        current = current.parentElement;
      }
      return true;
    }

    parseRTBActionPlanTable(table) {
      const headerRow = table.querySelector('thead tr, tr:first-child');
      if (!headerRow) return;

      // Get all header texts (handle nested elements and sort icons)
      const headerCells = headerRow.querySelectorAll('th, td');
      const headers = Array.from(headerCells).map(cell => {
        // Get just the text content, stripping sort arrows etc.
        return cell.textContent.replace(/[▲▼↑↓]/g, '').trim().toLowerCase();
      });

      console.log('[RTB Analyzer] Table headers:', headers.join(' | '));

      // Map RTB-specific columns
      const colMap = {
        actionPlanId: headers.findIndex(h => h.includes('action plan') || h.includes('plan id') || h.includes('action_plan')),
        path: headers.findIndex(h => h === 'path' || h.includes('path')),
        metric: headers.findIndex(h => h === 'metric' || (h.includes('metric') && !h.includes('action')) || h === 'kpi'),
        status: headers.findIndex(h => h === 'status'),
        category: headers.findIndex(h => h === 'category' || h.includes('category')),
        program: headers.findIndex(h => h === 'program' || h.includes('program')),
        initialValue: headers.findIndex(h => h.includes('initial') || h.includes('current') || h.includes('value') || h.includes('performance')),
        target: headers.findIndex(h => h === 'target' || h.includes('target') || h.includes('benchmark')),
        glidePathStatus: headers.findIndex(h => h.includes('glide path') || h.includes('glidepath status')),
        pctToGlidepath: headers.findIndex(h => h.includes('% to glidepath') || h.includes('glidepath')),
        pctToTarget: headers.findIndex(h => h.includes('% to target')),
        siteOwner: headers.findIndex(h => h.includes('site owner') || h.includes('owner')),
        startDate: headers.findIndex(h => h.includes('start date') || h.includes('date')),
      };

      // Need at least metric and some value columns to proceed
      if (colMap.metric < 0 && colMap.actionPlanId < 0) {
        console.log('[RTB Analyzer] Table rejected - no metric/actionPlan columns. Headers:', headers.join(' | '));
        // Fallback: if table has data rows with links, try using first column as metric name
        const bodyRows = table.querySelectorAll('tbody tr, tr:not(:first-child)');
        if (bodyRows.length > 0) {
          console.log('[RTB Analyzer] Attempting fallback parse with', bodyRows.length, 'rows');
          const firstRow = bodyRows[0];
          const firstCells = Array.from(firstRow.querySelectorAll('td')).map(c => c.textContent.trim());
          console.log('[RTB Analyzer] First data row:', firstCells.join(' | '));
          // Check if any cell contains a link to action-plan-details
          const rowLink = firstRow.querySelector('a[href*="action-plan"]');
          if (rowLink) {
            console.log('[RTB Analyzer] Found action-plan link in row:', rowLink.href);
            // Parse rows using positional approach: first col with link = metric name
            bodyRows.forEach(row => {
              const cells = row.querySelectorAll('td');
              if (cells.length < 2) return;
              const link = row.querySelector('a[href*="action-plan"]');
              const nameCell = link ? link.closest('td') || cells[0] : cells[0];
              const metricName = (link ? link.textContent.trim() : nameCell.textContent.trim());
              if (!metricName) return;
              this.metrics.push({
                name: metricName,
                actionPlanId: '',
                path: '',
                value: null,
                target: null,
                status: 'unknown',
                category: 'Other',
                program: '',
                glidePathStatus: '',
                pctToGlidepath: null,
                pctToTarget: null,
                percentOfTarget: null,
                siteOwner: '',
                startDate: '',
                planStatus: '',
                element: row,
                rawText: row.textContent.trim().substring(0, 300),
              });
            });
            console.log('[RTB Analyzer] Fallback parse got', this.metrics.length, 'metrics');
          }
        }
        return;
      }

      console.log('[RTB Analyzer] Column map:', colMap);

      // Parse data rows
      const bodyRows = table.querySelectorAll('tbody tr, tr:not(:first-child)');
      bodyRows.forEach(row => {
        const cells = row.querySelectorAll('td');
        if (cells.length < 3) return;

        const cellTexts = Array.from(cells).map(c => c.textContent.trim());

        const actionPlanId = colMap.actionPlanId >= 0 ? cellTexts[colMap.actionPlanId] : '';
        const path = colMap.path >= 0 ? cellTexts[colMap.path] : '';
        const metricName = colMap.metric >= 0 ? cellTexts[colMap.metric] : '';
        const status = colMap.status >= 0 ? cellTexts[colMap.status] : '';
        const category = colMap.category >= 0 ? cellTexts[colMap.category] : '';
        const program = colMap.program >= 0 ? cellTexts[colMap.program] : '';
        const initialValueText = colMap.initialValue >= 0 ? cellTexts[colMap.initialValue] : '';
        const targetText = colMap.target >= 0 ? cellTexts[colMap.target] : '';
        const glidePathStatus = colMap.glidePathStatus >= 0 ? cellTexts[colMap.glidePathStatus] : '';
        const pctToGlidepathText = colMap.pctToGlidepath >= 0 ? cellTexts[colMap.pctToGlidepath] : '';
        const pctToTargetText = colMap.pctToTarget >= 0 ? cellTexts[colMap.pctToTarget] : '';
        const siteOwner = colMap.siteOwner >= 0 ? cellTexts[colMap.siteOwner] : '';
        const startDate = colMap.startDate >= 0 ? cellTexts[colMap.startDate] : '';

        if (!metricName && !actionPlanId) return;

        const initialValue = this.parseNumericValue(initialValueText);
        const target = this.parseNumericValue(targetText);

        // Parse "% to Target" - format like "Behind (1 4.99%)" or "Ahead (10 4.78%)"
        const pctToTarget = this.parseRTBPercentage(pctToTargetText);
        const pctToGlidepath = this.parseRTBPercentage(pctToGlidepathText);

        // Determine status from Glide Path Status column
        const metricStatus = this.parseRTBStatus(glidePathStatus, pctToTarget);

        // Calculate actual % of target from initial value and target
        let percentOfTarget = null;
        if (initialValue !== null && target !== null && target !== 0) {
          percentOfTarget = ((initialValue / target) * 100).toFixed(1);
        }

        this.metrics.push({
          name: metricName,
          actionPlanId,
          path: path || '—',
          value: initialValue,
          target,
          status: metricStatus,
          category: category || 'Other',
          program: program || '',
          glidePathStatus: glidePathStatus,
          pctToGlidepath,
          pctToTarget,
          percentOfTarget,
          siteOwner,
          startDate,
          planStatus: status,
          element: row,
          rawText: row.textContent.trim().substring(0, 300),
        });
      });
    }

    parseRTBPercentage(text) {
      // Parses formats like "Behind (1 4.99%)", "Ahead (10 4.78%)", "Behind (7 4.46%)"
      // Also handles: "⊘ Behind (1 4.99%)" with icon prefixes, or multiline
      if (!text) return null;

      // Clean up the text - remove icons, extra whitespace, newlines
      const cleaned = text.replace(/[^\w\s().%\-]/g, '').replace(/\s+/g, ' ').trim();

      const match = cleaned.match(/(ahead|behind)\s*\((\d+)\s+([\d.]+)%?\)/i);
      if (match) {
        const direction = match[1].toLowerCase();
        const periods = parseInt(match[2], 10);
        const pct = parseFloat(match[3]);
        return {
          direction,
          periods,
          percentage: pct,
          raw: text.trim(),
          // Negative if behind, positive if ahead
          value: direction === 'ahead' ? pct : -pct,
        };
      }

      // Try alternate format: "Ahead (10\n4.78%)" with newline between number and percentage
      const altMatch = cleaned.match(/(ahead|behind)\s*\((\d+)\s*([\d.]+)%?\)/i);
      if (altMatch) {
        const direction = altMatch[1].toLowerCase();
        const periods = parseInt(altMatch[2], 10);
        const pct = parseFloat(altMatch[3]);
        return {
          direction, periods, percentage: pct,
          raw: text.trim(),
          value: direction === 'ahead' ? pct : -pct,
        };
      }

      // Try simpler format: just "AHEAD" or "BEHIND"
      if (cleaned.match(/ahead/i)) return { direction: 'ahead', periods: 0, percentage: 0, value: 0, raw: text.trim() };
      if (cleaned.match(/behind/i)) return { direction: 'behind', periods: 0, percentage: 0, value: 0, raw: text.trim() };

      return null;
    }

    parseRTBStatus(glidePathStatus, pctToTarget) {
      // Determine metric health status from RTB-specific fields
      const gpLower = (glidePathStatus || '').toLowerCase();
      const targetDir = pctToTarget?.direction || '';

      if (gpLower.includes('behind') && targetDir === 'behind') return 'critical';
      if (gpLower.includes('behind')) return 'warning';
      if (gpLower.includes('ahead') && targetDir === 'ahead') return 'exceeding';
      if (gpLower.includes('ahead')) return 'on-track';

      // Fallback to pctToTarget alone
      if (targetDir === 'behind') return 'warning';
      if (targetDir === 'ahead') return 'on-track';

      return 'unknown';
    }

    parseMetricCard(card) {
      const text = card.textContent.trim();
      if (!text || text.length > 500) return null;

      // Look for metric name, value, target/benchmark patterns
      const nameEl = card.querySelector(
        '[class*="name"], [class*="title"], [class*="label"], h3, h4, h5, strong, b'
      );
      const valueEl = card.querySelector(
        '[class*="value"], [class*="actual"], [class*="current"], [class*="score"]'
      );
      const targetEl = card.querySelector(
        '[class*="target"], [class*="benchmark"], [class*="goal"], [class*="expected"]'
      );

      if (!nameEl && !valueEl) return null;

      const name = nameEl ? nameEl.textContent.trim() : '';
      const valueText = valueEl ? valueEl.textContent.trim() : '';
      const targetText = targetEl ? targetEl.textContent.trim() : '';

      const value = this.parseNumericValue(valueText);
      const target = this.parseNumericValue(targetText);

      if (!name || value === null) return null;

      // Detect status from CSS classes or colors
      const status = this.detectStatus(card, value, target);
      const category = this.detectCategory(name, card);

      return {
        name, value, target, status, category,
        percentOfTarget: target ? ((value / target) * 100).toFixed(1) : null,
        element: card,
        rawText: text.substring(0, 200),
      };
    }

    parseMetricTable(table) {
      // Legacy fallback - only used if RTB-specific parsing finds nothing
      // The primary parser is parseRTBActionPlanTable above
      const rows = table.querySelectorAll('tr');
      if (rows.length < 2) return;

      const headers = [];
      const headerRow = rows[0];
      headerRow.querySelectorAll('th, td').forEach(cell => {
        headers.push(cell.textContent.trim().toLowerCase());
      });

      const nameIdx = headers.findIndex(h =>
        h.includes('metric') || h.includes('kpi') || h.includes('measure') || h.includes('name')
      );
      const valueIdx = headers.findIndex(h =>
        h.includes('actual') || h.includes('current') || h.includes('value') || h.includes('result')
      );
      const targetIdx = headers.findIndex(h =>
        h.includes('target') || h.includes('benchmark') || h.includes('goal') || h.includes('plan')
      );

      if (nameIdx < 0 && valueIdx < 0) return;

      for (let i = 1; i < rows.length; i++) {
        const cells = rows[i].querySelectorAll('td');
        if (cells.length < 2) continue;

        const name = nameIdx >= 0 ? cells[nameIdx]?.textContent.trim() : '';
        const valueText = valueIdx >= 0 ? cells[valueIdx]?.textContent.trim() : '';
        const targetText = targetIdx >= 0 ? cells[targetIdx]?.textContent.trim() : '';

        const value = this.parseNumericValue(valueText);
        const target = this.parseNumericValue(targetText);

        if (!name || value === null) continue;

        this.metrics.push({
          name, value, target,
          status: this.detectStatus(rows[i], value, target),
          category: this.detectCategory(name, rows[i]),
          percentOfTarget: target ? ((value / target) * 100).toFixed(1) : null,
          element: rows[i],
          rawText: rows[i].textContent.trim().substring(0, 200),
        });
      }
    }

    parseMetricRow(row) {
      const text = row.textContent.trim();
      if (!text || text.length > 300 || text.length < 5) return null;

      // Pattern: "Metric Name: 85.2% (Target: 95%)" or similar
      const patterns = [
        /^(.+?):\s*([\d.]+%?)\s*(?:\((?:target|goal|benchmark):\s*([\d.]+%?)\))?/i,
        /^(.+?)\s+([\d.]+%?)\s*\/\s*([\d.]+%?)/,
      ];

      for (const pattern of patterns) {
        const match = text.match(pattern);
        if (match) {
          const name = match[1].trim();
          const value = this.parseNumericValue(match[2]);
          const target = match[3] ? this.parseNumericValue(match[3]) : null;
          if (name && value !== null) {
            return {
              name, value, target,
              status: this.detectStatus(row, value, target),
              category: this.detectCategory(name, row),
              percentOfTarget: target ? ((value / target) * 100).toFixed(1) : null,
              element: row,
              rawText: text.substring(0, 200),
            };
          }
        }
      }
      return null;
    }

    extractComments() {
      // Strategy 1: Look for RTB "Action Plan Comments" tab content
      // The RTB page has tabs: Current Action Plans | Future Action Plans | Action Plan Comments | Resolved Action Plans
      const allText = document.body.textContent;

      // Look for comment sections specific to RTB
      const commentContainers = document.querySelectorAll(
        '[class*="comment"], [class*="note"], [class*="annotation"],' +
        '[class*="remark"], [class*="feedback"], [class*="update"],' +
        '[class*="discussion"], [class*="thread"], [class*="message"],' +
        '[role="comment"], [data-comment], [data-note]'
      );
      commentContainers.forEach(el => {
        if (!this.isElementVisible(el)) return;
        const comment = this.parseComment(el);
        if (comment) this.comments.push(comment);
      });

      // Strategy 2: Look for comment tables (RTB may show comments in table format)
      const tables = document.querySelectorAll('table');
      tables.forEach(table => {
        if (!this.isElementVisible(table)) return;
        const headers = Array.from(table.querySelectorAll('thead th, tr:first-child th, tr:first-child td'))
          .map(c => c.textContent.trim().toLowerCase());

        // Check if this is a comments table
        const isCommentTable = headers.some(h =>
          h.includes('comment') || h.includes('note') || h.includes('update') || h.includes('description')
        );
        if (!isCommentTable) return;

        const commentIdx = headers.findIndex(h => h.includes('comment') || h.includes('note') || h.includes('description') || h.includes('update'));
        const authorIdx = headers.findIndex(h => h.includes('author') || h.includes('user') || h.includes('by') || h.includes('owner'));
        const dateIdx = headers.findIndex(h => h.includes('date') || h.includes('time') || h.includes('created'));
        const metricIdx = headers.findIndex(h => h.includes('metric') || h.includes('plan') || h.includes('action'));

        const rows = table.querySelectorAll('tbody tr, tr:not(:first-child)');
        rows.forEach(row => {
          const cells = Array.from(row.querySelectorAll('td')).map(c => c.textContent.trim());
          const text = commentIdx >= 0 ? cells[commentIdx] : cells.find(c => c.length > 20) || '';
          if (!text || text.length < 5) return;

          this.comments.push({
            text,
            author: authorIdx >= 0 ? cells[authorIdx] : null,
            timestamp: dateIdx >= 0 ? cells[dateIdx] : null,
            metric: metricIdx >= 0 ? cells[metricIdx] : this.findMetricContext(row),
            element: row,
            themes: this.extractThemes(text),
          });
        });
      });

      // Strategy 3: Look for expandable/collapsible comment sections within action plan rows
      const expandedSections = document.querySelectorAll(
        '[class*="expand"], [class*="detail"], [class*="collapse"],' +
        '[class*="accordion"], [class*="panel-body"], [aria-expanded]'
      );
      expandedSections.forEach(el => {
        if (!this.isElementVisible(el)) return;
        const text = el.textContent.trim();
        if (text.length < 10 || text.length > 2000) return;
        // Skip if it looks like the main table data
        if (text.match(/^(Open|Closed|Variation Reduction|Road To Benchmark)/)) return;

        const comment = this.parseComment(el);
        if (comment) this.comments.push(comment);
      });

      // Deduplicate comments
      const seen = new Set();
      this.comments = this.comments.filter(c => {
        const key = c.text.substring(0, 100);
        if (seen.has(key)) return false;
        seen.add(key);
        return true;
      });
    }

    parseComment(el) {
      const text = el.textContent.trim();
      if (!text || text.length < 5 || text.length > 2000) return null;

      // Try to find associated metric
      const metricContext = this.findMetricContext(el);

      // Try to find author and timestamp
      const author = this.findAuthor(el);
      const timestamp = this.findTimestamp(el);

      return {
        text, author, timestamp,
        metric: metricContext,
        element: el,
        themes: this.extractThemes(text),
      };
    }

    parseCommentStructure(el) {
      const authorEl = el.querySelector('[class*="author"], [class*="user"], [class*="name"]');
      const textEl = el.querySelector('[class*="body"], [class*="content"], [class*="text"], p');
      const timeEl = el.querySelector('[class*="time"], [class*="date"], time');

      if (!textEl) return null;
      const text = textEl.textContent.trim();
      if (!text || text.length < 5) return null;

      return {
        text,
        author: authorEl ? authorEl.textContent.trim() : 'Unknown',
        timestamp: timeEl ? timeEl.textContent.trim() : null,
        metric: this.findMetricContext(el),
        element: el,
        themes: this.extractThemes(text),
      };
    }

    extractHistoricalData() {
      // Look for time-series data in tables or charts
      const tables = document.querySelectorAll('table');
      tables.forEach(table => {
        const headers = Array.from(table.querySelectorAll('tr:first-child th, tr:first-child td'))
          .map(c => c.textContent.trim());

        // Check if headers contain date/week/period patterns
        const dateHeaders = headers.filter(h =>
          h.match(/wk|week|w\d|period|p\d|month|q[1-4]|\d{1,2}\/\d{1,2}/i)
        );

        if (dateHeaders.length >= 2) {
          const rows = table.querySelectorAll('tr');
          for (let i = 1; i < rows.length; i++) {
            const cells = Array.from(rows[i].querySelectorAll('td'));
            if (cells.length < 2) continue;
            const metricName = cells[0]?.textContent.trim();
            const values = cells.slice(1).map(c => this.parseNumericValue(c.textContent.trim()));
            if (metricName && values.some(v => v !== null)) {
              this.historicalData.push({
                metric: metricName,
                periods: dateHeaders,
                values: values,
              });
            }
          }
        }
      });

      // Also check for chart data attributes
      const charts = document.querySelectorAll('[data-values], [data-series], canvas, svg');
      charts.forEach(chart => {
        const dataAttr = chart.getAttribute('data-values') || chart.getAttribute('data-series');
        if (dataAttr) {
          try {
            const parsed = JSON.parse(dataAttr);
            if (Array.isArray(parsed)) {
              this.historicalData.push({
                metric: chart.getAttribute('data-label') || 'Chart Data',
                values: parsed,
              });
            }
          } catch (e) { /* ignore parse errors */ }
        }
      });
    }

    // ---- Helper Methods ----

    parseNumericValue(text) {
      if (!text) return null;
      const cleaned = text.replace(/[,%$]/g, '').trim();
      const num = parseFloat(cleaned);
      return isNaN(num) ? null : num;
    }

    detectStatus(element, value, target) {
      // Check for RTB-specific status text in the element
      const text = (element.textContent || '').toLowerCase();
      if (text.includes('behind')) return 'warning';
      if (text.includes('ahead')) return 'on-track';

      // Check CSS classes for status indicators
      const classes = element.className || '';
      if (classes.match(/critical|red|danger|fail/i)) return 'critical';
      if (classes.match(/warning|yellow|amber|caution/i)) return 'warning';
      if (classes.match(/success|green|good|pass|on.?track/i)) return 'on-track';
      if (classes.match(/exceed|blue|excellent/i)) return 'exceeding';

      // Check for colored icons/spans within the element
      const icons = element.querySelectorAll('[style*="color"], [class*="red"], [class*="green"]');
      for (const icon of icons) {
        const style = icon.getAttribute('style') || '';
        const cls = icon.className || '';
        if (style.includes('red') || cls.includes('red') || cls.includes('danger')) return 'critical';
        if (style.includes('green') || cls.includes('green') || cls.includes('success')) return 'on-track';
      }

      // Calculate from value vs target
      if (value !== null && target) {
        const pct = (value / target) * 100;
        if (pct < CONFIG.CRITICAL_THRESHOLD) return 'critical';
        if (pct < CONFIG.WARNING_THRESHOLD) return 'warning';
        if (pct >= 100) return 'exceeding';
        return 'on-track';
      }

      return 'unknown';
    }

    detectCategory(name, element) {
      // For RTB, the category comes directly from the table column
      // This is a fallback for non-table metrics
      const text = (name + ' ' + (element?.textContent || '')).toLowerCase();
      const categories = {
        'Safety': /safety|injury|incident|osha|recordable|trir|dart|first.?aid/,
        'Quality': /quality|defect|dpmo|error|accuracy|compliance|audit/,
        'Productivity': /productivity|rate|uph|jph|cph|throughput|output|units|cplh/i,
        'Delivery': /delivery|on.?time|otd|promise|sla|speed|cycle/,
        'Cost': /cost|cpt|spend|budget|variance|savings/,
        'People': /people|engagement|attrition|retention|headcount|attendance/,
        'Inventory': /inventory|shrink|phantom|misplace|concession/,
      };
      for (const [cat, pattern] of Object.entries(categories)) {
        if (text.match(pattern)) return cat;
      }
      return 'Other';
    }

    findMetricContext(el) {
      // Walk up the DOM to find which metric this comment belongs to
      let parent = el.parentElement;
      for (let depth = 0; depth < 5 && parent; depth++) {
        const heading = parent.querySelector('h1, h2, h3, h4, h5, [class*="title"], [class*="name"]');
        if (heading) {
          const text = heading.textContent.trim();
          if (text.length < 80 && text.length > 2) return text;
        }
        parent = parent.parentElement;
      }
      // Check siblings
      let prev = el.previousElementSibling;
      for (let i = 0; i < 3 && prev; i++) {
        const text = prev.textContent.trim();
        if (text.length < 80 && text.length > 2 && !text.includes('\n')) return text;
        prev = prev.previousElementSibling;
      }
      return null;
    }

    findAuthor(el) {
      const authorEl = el.querySelector('[class*="author"], [class*="user"], [class*="by"]');
      if (authorEl) return authorEl.textContent.trim();
      // Check for "- username" or "by username" patterns
      const text = el.textContent;
      const match = text.match(/(?:^|\n)\s*[-–—]\s*(\w+(?:\s\w+)?)\s*$/);
      if (match) return match[1];
      return null;
    }

    findTimestamp(el) {
      const timeEl = el.querySelector('time, [class*="time"], [class*="date"], [datetime]');
      if (timeEl) return timeEl.getAttribute('datetime') || timeEl.textContent.trim();
      const text = el.textContent;
      const match = text.match(/(\d{1,2}\/\d{1,2}\/\d{2,4}|\d{4}-\d{2}-\d{2}|(?:Jan|Feb|Mar|Apr|May|Jun|Jul|Aug|Sep|Oct|Nov|Dec)\s+\d{1,2})/i);
      return match ? match[1] : null;
    }

    extractThemes(text) {
      const themes = [];
      const themePatterns = {
        'staffing': /staff|headcount|labor|hiring|short.?handed|understaffed/i,
        'equipment': /equipment|machine|scanner|device|broken|maintenance/i,
        'process': /process|workflow|procedure|sop|standard work/i,
        'training': /train|coach|learn|new hire|ramp|onboard/i,
        'volume': /volume|demand|peak|surge|capacity/i,
        'quality-issue': /defect|error|mistake|rework|reject/i,
        'improvement': /improve|better|progress|gain|increase/i,
        'blocker': /block|prevent|barrier|obstacle|issue|problem/i,
        'action-item': /action|todo|follow.?up|next step|plan/i,
        'positive': /great|excellent|good job|well done|success|achieve/i,
      };
      for (const [theme, pattern] of Object.entries(themePatterns)) {
        if (text.match(pattern)) themes.push(theme);
      }
      return themes;
    }
  }

  // ============================================================
  // ANALYSIS ENGINE - Trend detection, recommendations, summaries
  // ============================================================
  class RTBAnalysisEngine {
    constructor(data) {
      this.metrics = data.metrics || [];
      this.comments = data.comments || [];
      this.historical = data.historical || [];
    }

    // ---- Performance Analysis ----
    analyzePerformance() {
      return this.metrics.map(metric => {
        const pct = metric.percentOfTarget ? parseFloat(metric.percentOfTarget) : null;
        const trend = this.calculateTrend(metric.name);
        const recommendation = this.generateRecommendation(metric, trend);

        return {
          ...metric,
          trend,
          recommendation,
          riskLevel: this.assessRisk(metric, trend),
        };
      });
    }

    // ---- Trend Analysis ----
    calculateTrend(metricName) {
      const history = this.historical.find(h =>
        h.metric.toLowerCase().includes(metricName.toLowerCase()) ||
        metricName.toLowerCase().includes(h.metric.toLowerCase())
      );

      if (!history || !history.values || history.values.length < 2) {
        return { direction: 'unknown', magnitude: 0, periods: 0, values: [] };
      }

      const values = history.values.filter(v => v !== null);
      if (values.length < 2) return { direction: 'unknown', magnitude: 0, periods: 0, values };

      // Calculate linear regression slope
      const n = values.length;
      const xMean = (n - 1) / 2;
      const yMean = values.reduce((s, v) => s + v, 0) / n;
      let numerator = 0, denominator = 0;
      for (let i = 0; i < n; i++) {
        numerator += (i - xMean) * (values[i] - yMean);
        denominator += (i - xMean) * (i - xMean);
      }
      const slope = denominator !== 0 ? numerator / denominator : 0;

      // Determine direction and magnitude
      const avgValue = yMean || 1;
      const slopePercent = (slope / avgValue) * 100;
      let direction = 'flat';
      if (slopePercent > 2) direction = 'improving';
      else if (slopePercent < -2) direction = 'declining';

      // Check recent momentum (last 2 vs previous 2)
      let momentum = 'steady';
      if (values.length >= 4) {
        const recent = (values[values.length - 1] + values[values.length - 2]) / 2;
        const prior = (values[values.length - 3] + values[values.length - 4]) / 2;
        const change = ((recent - prior) / (prior || 1)) * 100;
        if (change > 5) momentum = 'accelerating';
        else if (change < -5) momentum = 'decelerating';
      }

      return {
        direction, momentum,
        magnitude: Math.abs(slopePercent).toFixed(1),
        slope: slope.toFixed(2),
        periods: values.length,
        values,
        recentChange: values.length >= 2
          ? ((values[values.length - 1] - values[values.length - 2]) / (values[values.length - 2] || 1) * 100).toFixed(1)
          : 0,
      };
    }

    analyzeTrendsSummary() {
      const trends = [];
      this.metrics.forEach(metric => {
        const trend = this.calculateTrend(metric.name);
        if (trend.direction !== 'unknown') {
          trends.push({ metric: metric.name, category: metric.category, ...trend });
        }
      });

      // Group by direction
      const improving = trends.filter(t => t.direction === 'improving');
      const declining = trends.filter(t => t.direction === 'declining');
      const flat = trends.filter(t => t.direction === 'flat');

      return { improving, declining, flat, all: trends };
    }

    // ---- Recommendations ----
    generateRecommendation(metric, trend) {
      const pct = metric.percentOfTarget ? parseFloat(metric.percentOfTarget) : null;
      const pctToTarget = metric.pctToTarget;
      const pctToGlidepath = metric.pctToGlidepath;
      const recs = [];

      // Use RTB-specific "behind/ahead" status for recommendations
      if (pctToTarget && pctToTarget.direction === 'behind') {
        const behindPct = pctToTarget.percentage;
        const periods = pctToTarget.periods;

        if (behindPct > 8 || periods >= 7) {
          recs.push({
            priority: 'Critical',
            title: `${metric.name} is significantly behind target`,
            action: `Behind target by ${behindPct}% for ${periods} period(s). Current: ${metric.value}, Target: ${metric.target}. Requires immediate escalation.`,
            suggestions: [
              'Conduct root cause analysis with process owners and site leadership',
              'Establish daily metric review cadence with action tracking',
              'Identify structural barriers vs execution gaps',
              'Consider resource reallocation or process redesign',
              'Escalate to senior leadership if systemic issues identified',
            ],
          });
        } else if (behindPct > 4) {
          recs.push({
            priority: 'High',
            title: `${metric.name} needs focused improvement plan`,
            action: `Behind target by ${behindPct}% for ${periods} period(s). Gap is material but recoverable with focused effort.`,
            suggestions: [
              'Review action plan effectiveness — are current actions driving improvement?',
              'Set weekly milestones to close the gap incrementally',
              'Benchmark against similar paths/sites that are meeting target',
              'Ensure adequate staffing and resource allocation for this metric',
              'Update action plan with specific countermeasures',
            ],
          });
        } else {
          recs.push({
            priority: 'Medium',
            title: `${metric.name} is slightly behind target`,
            action: `Behind by ${behindPct}% for ${periods} period(s). Small adjustments should close the gap.`,
            suggestions: [
              'Identify the 1-2 specific drivers keeping performance below target',
              'Implement targeted micro-improvements',
              'Monitor daily to ensure sustained progress toward glidepath',
            ],
          });
        }
      } else if (pctToTarget && pctToTarget.direction === 'ahead') {
        recs.push({
          priority: 'Maintain',
          title: `${metric.name} is ahead of target`,
          action: `Ahead by ${pctToTarget.percentage}% for ${pctToTarget.periods} period(s). Focus on sustainability.`,
          suggestions: [
            'Document what is working well for knowledge sharing across paths',
            'Set stretch goals to continue improvement momentum',
            'Share best practices with metrics that are behind',
          ],
        });
      } else if (pct !== null && pct < CONFIG.CRITICAL_THRESHOLD) {
        // Fallback to generic % of target analysis
        recs.push({
          priority: 'Critical',
          title: `${metric.name} is significantly below benchmark`,
          action: `Currently at ${pct}% of target. Requires immediate intervention.`,
          suggestions: [
            'Conduct deep-dive with process owners',
            'Establish daily tracking cadence',
            'Identify quick wins to close gap within 1-2 weeks',
          ],
        });
      }

      // Glidepath-specific recommendations
      if (pctToGlidepath && pctToGlidepath.direction === 'behind' && pctToGlidepath.percentage > 5) {
        recs.push({
          priority: 'Watch',
          title: `${metric.name} is falling off glidepath`,
          action: `Behind glidepath by ${pctToGlidepath.percentage}%. The planned improvement trajectory is not being met.`,
          suggestions: [
            'Review if the glidepath assumptions are still valid',
            'Identify what changed since the action plan was created',
            'Adjust glidepath if external factors have shifted, or intensify actions',
            'Consider whether the action plan needs to be revised',
          ],
        });
      }

      // Add trend-based recommendations
      if (trend.direction === 'declining') {
        recs.push({
          priority: 'Watch',
          title: `Declining trend detected for ${metric.name}`,
          action: `Performance declining at ${trend.magnitude}% per period. ${trend.momentum === 'decelerating' ? 'Decline is accelerating.' : ''}`,
          suggestions: [
            'Investigate what changed in recent periods',
            'Check for seasonal patterns vs structural issues',
            'Engage frontline teams for ground-truth insights',
          ],
        });
      } else if (trend.direction === 'improving' && trend.momentum === 'accelerating') {
        recs.push({
          priority: 'Positive',
          title: `Strong improvement momentum on ${metric.name}`,
          action: `Improving at ${trend.magnitude}% per period with accelerating gains.`,
          suggestions: [
            'Identify and reinforce the drivers of improvement',
            'Project when benchmark will be achieved at current rate',
            'Celebrate progress with the team',
          ],
        });
      }

      return recs;
    }

    assessRisk(metric, trend) {
      const pct = metric.percentOfTarget ? parseFloat(metric.percentOfTarget) : 100;
      let risk = 0;

      // Distance from target
      if (pct < CONFIG.CRITICAL_THRESHOLD) risk += 3;
      else if (pct < CONFIG.WARNING_THRESHOLD) risk += 2;
      else if (pct < CONFIG.ON_TRACK_THRESHOLD) risk += 1;

      // Trend direction
      if (trend.direction === 'declining') risk += 2;
      if (trend.momentum === 'decelerating') risk += 1;

      if (risk >= 4) return 'high';
      if (risk >= 2) return 'medium';
      return 'low';
    }

    // ---- Comment Analysis ----
    analyzeComments() {
      // Group comments by metric
      const byMetric = {};
      this.comments.forEach(comment => {
        const key = comment.metric || 'General';
        if (!byMetric[key]) byMetric[key] = [];
        byMetric[key].push(comment);
      });

      const summaries = {};
      for (const [metric, comments] of Object.entries(byMetric)) {
        summaries[metric] = this.summarizeComments(comments, metric);
      }
      return summaries;
    }

    summarizeComments(comments, metricName) {
      if (!comments || comments.length === 0) return null;

      // Extract all themes
      const allThemes = {};
      comments.forEach(c => {
        (c.themes || []).forEach(theme => {
          allThemes[theme] = (allThemes[theme] || 0) + 1;
        });
      });

      // Sort themes by frequency
      const topThemes = Object.entries(allThemes)
        .sort((a, b) => b[1] - a[1])
        .slice(0, 5)
        .map(([theme, count]) => ({ theme, count }));

      // Identify key action items
      const actionItems = comments
        .filter(c => c.themes && c.themes.includes('action-item'))
        .map(c => c.text.substring(0, 150));

      // Identify blockers
      const blockers = comments
        .filter(c => c.themes && c.themes.includes('blocker'))
        .map(c => c.text.substring(0, 150));

      // Identify positive callouts
      const positives = comments
        .filter(c => c.themes && c.themes.includes('positive'))
        .map(c => c.text.substring(0, 150));

      // Generate narrative summary
      const narrative = this.generateNarrative(comments, topThemes, metricName);

      // Sentiment analysis (simple)
      const sentimentScores = comments.map(c => this.scoreSentiment(c.text));
      const avgSentiment = sentimentScores.reduce((s, v) => s + v, 0) / sentimentScores.length;

      return {
        metricName,
        commentCount: comments.length,
        topThemes,
        actionItems: actionItems.slice(0, 5),
        blockers: blockers.slice(0, 5),
        positives: positives.slice(0, 3),
        narrative,
        sentiment: avgSentiment > 0.3 ? 'positive' : avgSentiment < -0.3 ? 'negative' : 'neutral',
        sentimentScore: avgSentiment.toFixed(2),
        recentComments: comments.slice(-3).map(c => ({
          text: c.text.substring(0, 200),
          author: c.author,
          timestamp: c.timestamp,
        })),
      };
    }

    generateNarrative(comments, topThemes, metricName) {
      const parts = [];
      parts.push(`${comments.length} comment(s) found for ${metricName}.`);

      if (topThemes.length > 0) {
        const themeNames = topThemes.slice(0, 3).map(t => t.theme).join(', ');
        parts.push(`Primary themes: ${themeNames}.`);
      }

      const blockerCount = comments.filter(c => c.themes?.includes('blocker')).length;
      if (blockerCount > 0) {
        parts.push(`${blockerCount} comment(s) mention blockers or issues.`);
      }

      const actionCount = comments.filter(c => c.themes?.includes('action-item')).length;
      if (actionCount > 0) {
        parts.push(`${actionCount} action item(s) identified.`);
      }

      const improvementCount = comments.filter(c => c.themes?.includes('improvement')).length;
      if (improvementCount > 0) {
        parts.push(`${improvementCount} comment(s) note improvements or progress.`);
      }

      return parts.join(' ');
    }

    scoreSentiment(text) {
      const positive = /good|great|excellent|improve|better|success|achieve|progress|gain|strong|well/gi;
      const negative = /bad|poor|fail|decline|worse|issue|problem|miss|behind|concern|risk|gap/gi;
      const posCount = (text.match(positive) || []).length;
      const negCount = (text.match(negative) || []).length;
      const total = posCount + negCount;
      if (total === 0) return 0;
      return (posCount - negCount) / total;
    }
  }

    async fetchAllTrendsWithProgress(progressElId) {
      const allTrends = [];
      const metrics = this.analysisResults;
      if (metrics.length === 0) return allTrends;

      const progressEl = document.getElementById(progressElId);

      for (let i = 0; i < metrics.length; i++) {
        const metric = metrics[i];
        if (progressEl) {
          progressEl.textContent = `Fetching weekly data for ${metric.name} (${i + 1}/${metrics.length})...`;
        }

        const link = this.findMetricLink(metric);
        if (!link) continue;

        try {
          const trendData = await this.fetchTrendViaIframe(link, metric.name);
          if (trendData && trendData.length > 0) {
            allTrends.push({ metric: metric.name, path: metric.path, data: trendData });
          }
        } catch (err) {
          console.warn('[RTB Analyzer] Summary: failed to fetch trends for', metric.name, err.message);
        }

        if (i < metrics.length - 1) {
          await new Promise(r => setTimeout(r, 500));
        }
      }

      if (progressEl) progressEl.textContent = 'Done — generating summary...';
      return allTrends;
    }

    generateWeeklySummary(metrics, actions, trends) {
      const total = metrics.length;
      const behind = metrics.filter(m => m.status === 'critical' || m.status === 'warning');
      const ahead = metrics.filter(m => m.status === 'on-track' || m.status === 'exceeding');
      const critical = metrics.filter(m => m.status === 'critical');

      // WoW changes from trend data — includes variance and multi-week context
      const wowChanges = [];
      metrics.forEach(m => {
        const trend = trends.find(t => t.metric === m.name);
        if (trend && trend.data && trend.data.length >= 2) {
          // Sort chronologically (oldest first)
          const sorted = [...trend.data].sort((a, b) => (a.date || '').localeCompare(b.date || ''));
          const latest = sorted[sorted.length - 1];
          const previous = sorted[sorted.length - 2];
          if (latest && previous && latest.performance != null && previous.performance != null) {
            const change = latest.performance - previous.performance;
            const pctChange = previous.performance !== 0 ? ((change / previous.performance) * 100) : 0;
            const target = latest.target ?? m.target;
            const varianceFromTarget = (target != null && latest.performance != null) ? latest.performance - target : null;
            const prevVariance = (target != null && previous.performance != null) ? previous.performance - target : null;

            // 4-week trend (if available)
            const last4 = sorted.slice(-4);
            const fourWeekTrend = last4.length >= 3 ? last4.map(d => d.performance) : null;
            const fourWeekDirection = fourWeekTrend && fourWeekTrend.length >= 3
              ? (fourWeekTrend[fourWeekTrend.length - 1] > fourWeekTrend[0] ? 'improving' : fourWeekTrend[fourWeekTrend.length - 1] < fourWeekTrend[0] ? 'declining' : 'flat')
              : null;

            // Consecutive weeks behind/ahead
            let consecutiveBehind = 0;
            let consecutiveAhead = 0;
            if (target != null) {
              for (let i = sorted.length - 1; i >= 0; i--) {
                if (sorted[i].performance < target) consecutiveBehind++;
                else break;
              }
              for (let i = sorted.length - 1; i >= 0; i--) {
                if (sorted[i].performance >= target) consecutiveAhead++;
                else break;
              }
            }

            wowChanges.push({
              name: m.name,
              current: latest.performance,
              previous: previous.performance,
              change,
              pctChange,
              target,
              varianceFromTarget,
              prevVariance,
              status: m.status,
              date: latest.date,
              prevDate: previous.date,
              fourWeekTrend,
              fourWeekDirection,
              consecutiveBehind,
              consecutiveAhead,
              glidepath: latest.glidepath,
              weeklyData: sorted.slice(-6), // Last 6 weeks for sparkline context
              owner: m.owner || m.siteOwner,
              category: m.category,
            });
          }
        } else if (m.wowTrend != null) {
          // Fallback to API WoW trend field
          const target = m.target;
          const current = m.value;
          const previous = current != null && m.wowTrend != null ? current - m.wowTrend : null;
          const varianceFromTarget = (target != null && current != null) ? current - target : null;
          wowChanges.push({
            name: m.name,
            current,
            previous,
            change: m.wowTrend,
            pctChange: previous && previous !== 0 ? ((m.wowTrend / previous) * 100) : 0,
            target,
            varianceFromTarget,
            prevVariance: previous != null && target != null ? previous - target : null,
            status: m.status,
            fourWeekDirection: null,
            consecutiveBehind: 0,
            consecutiveAhead: 0,
            weeklyData: null,
            owner: m.owner || m.siteOwner,
            category: m.category,
          });
        }
      });

      // Highlights: metrics that improved most (closing gap to target)
      const highlights = wowChanges
        .filter(w => {
          // Improving = moving closer to target or positive change when above target
          if (w.target != null && w.varianceFromTarget != null && w.prevVariance != null) {
            return Math.abs(w.varianceFromTarget) < Math.abs(w.prevVariance);
          }
          return w.change > 0;
        })
        .sort((a, b) => {
          // Sort by how much gap closed
          const aGapClosed = a.prevVariance != null && a.varianceFromTarget != null ? Math.abs(a.prevVariance) - Math.abs(a.varianceFromTarget) : a.change;
          const bGapClosed = b.prevVariance != null && b.varianceFromTarget != null ? Math.abs(b.prevVariance) - Math.abs(b.varianceFromTarget) : b.change;
          return bGapClosed - aGapClosed;
        })
        .slice(0, 8);

      // Lowlights: metrics that declined most (growing gap to target)
      const lowlights = wowChanges
        .filter(w => {
          if (w.target != null && w.varianceFromTarget != null && w.prevVariance != null) {
            return Math.abs(w.varianceFromTarget) > Math.abs(w.prevVariance);
          }
          return w.change < 0;
        })
        .sort((a, b) => {
          const aGapGrown = a.varianceFromTarget != null && a.prevVariance != null ? Math.abs(a.varianceFromTarget) - Math.abs(a.prevVariance) : Math.abs(a.change);
          const bGapGrown = b.varianceFromTarget != null && b.prevVariance != null ? Math.abs(b.varianceFromTarget) - Math.abs(b.prevVariance) : Math.abs(b.change);
          return bGapGrown - aGapGrown;
        })
        .slice(0, 8);

      // Action summary
      const openActions = actions.filter(a => a.status?.match(/open|not started|in progress/i));
      const completedActions = actions.filter(a => a.status?.match(/complete|closed|done/i));
      const overdueActions = actions.filter(a => {
        if (!a.dueDate) return false;
        try { return new Date(a.dueDate) < new Date(); } catch (e) { return false; }
      });

      // Overall progress: how many metrics are trending toward target
      const improving = wowChanges.filter(w => {
        if (w.target == null || w.current == null) return false;
        const gapNow = Math.abs(w.target - w.current);
        const gapBefore = Math.abs(w.target - (w.previous ?? w.current));
        return gapNow < gapBefore;
      });
      const declining = wowChanges.filter(w => {
        if (w.target == null || w.current == null) return false;
        const gapNow = Math.abs(w.target - w.current);
        const gapBefore = Math.abs(w.target - (w.previous ?? w.current));
        return gapNow > gapBefore;
      });

      return {
        total, behind, ahead, critical, highlights, lowlights,
        wowChanges, openActions, completedActions, overdueActions,
        improving, declining, actions, trends,
      };
    }

    renderMetricSummaryCard(w, actions, cardType) {
      const changeStr = w.change > 0 ? `+${w.change.toFixed(2)}` : w.change.toFixed(2);
      const trendClass = w.change > 0 ? 'rtb-trend-up' : w.change < 0 ? 'rtb-trend-down' : 'rtb-trend-flat';

    renderMiniSparkline(weeklyData, target) {
      if (!weeklyData || weeklyData.length < 2) return '';
      const values = weeklyData.map(d => d.performance).filter(v => v != null);
      if (values.length < 2) return '';

      const min = Math.min(...values, target || Infinity);
      const max = Math.max(...values, target || -Infinity);
      const range = max - min || 1;
      const height = 40;
      const width = 80;
      const step = width / (values.length - 1);

      let path = '';
      values.forEach((v, i) => {
        const x = i * step;
        const y = height - ((v - min) / range) * height;
        path += i === 0 ? `M${x},${y}` : ` L${x},${y}`;
      });

      let targetLine = '';
      if (target != null && target >= min && target <= max) {
        const ty = height - ((target - min) / range) * height;
        targetLine = `<line x1="0" y1="${ty}" x2="${width}" y2="${ty}" stroke="#ff9900" stroke-width="1" stroke-dasharray="3,2" opacity="0.6"/>`;
      }

      return `<svg width="${width}" height="${height}" style="display:block;">
        ${targetLine}
        <path d="${path}" fill="none" stroke="#3498db" stroke-width="1.5"/>
        <circle cx="${(values.length - 1) * step}" cy="${height - ((values[values.length - 1] - min) / range) * height}" r="2.5" fill="#3498db"/>
      </svg>`;
    }

    summarizeActionText(text, maxWords = 50) {
      // Smart summarization: condenses action description to ~maxWords while keeping key info
      if (!text) return '';
      const words = text.split(/\s+/);
      if (words.length <= maxWords) return text;

      // Strategy: extract key sentences/clauses with actionable info
      // Split into sentences
      const sentences = text.split(/(?<=[.!;])\s+/).filter(s => s.trim().length > 0);

      // Score each sentence for importance (action verbs, metrics, thresholds, roles)
      const importancePatterns = /\b(must|will|requires?|ensures?|prevents?|triggers?|target|threshold|baseline|above|below|maintain|implement|track|enforce|reduce|increase|designated|mandatory|immediately|daily|hourly|weekly)\b/gi;
      const numericPattern = /\d+%?|\d+\s*(or more|pallets?|lines?|hours?|minutes?)/i;

      const scored = sentences.map(s => {
        let score = 0;
        const importanceMatches = s.match(importancePatterns);
        if (importanceMatches) score += importanceMatches.length * 2;
        if (s.match(numericPattern)) score += 3; // Numeric thresholds are key
        if (s === sentences[0]) score += 2; // First sentence usually states the what
        return { text: s, score, words: s.split(/\s+/).length };
      });

      // Sort by score descending, then take sentences until we hit the word limit
      scored.sort((a, b) => b.score - a.score);

      let result = [];
      let wordCount = 0;

      for (const s of scored) {
        if (wordCount + s.words <= maxWords) {
          result.push(s);
          wordCount += s.words;
        } else if (wordCount === 0) {
          // First sentence exceeds limit — truncate it smartly at a clause boundary
          const clauses = s.text.split(/,\s*/);
          let truncated = '';
          let truncWords = 0;
          for (const clause of clauses) {
            const cWords = clause.split(/\s+/).length;
            if (truncWords + cWords <= maxWords) {
              truncated += (truncated ? ', ' : '') + clause;
              truncWords += cWords;
            } else break;
          }
          if (!truncated) {
            // No clause boundary — just take maxWords
            truncated = s.text.split(/\s+/).slice(0, maxWords).join(' ');
          }
          result.push({ text: truncated.replace(/[,;]$/, '.'), score: s.score, words: truncated.split(/\s+/).length });
          wordCount += result[result.length - 1].words;
          break;
        }
      }

      // Re-order result to match original sentence order
      const originalOrder = sentences.map(s => s.trim());
      result.sort((a, b) => {
        const aIdx = originalOrder.findIndex(o => o.startsWith(a.text.substring(0, 20)));
        const bIdx = originalOrder.findIndex(o => o.startsWith(b.text.substring(0, 20)));
        return (aIdx === -1 ? 999 : aIdx) - (bIdx === -1 ? 999 : bIdx);
      });

      return result.map(r => r.text).join(' ').replace(/\s+/g, ' ').trim();
    }

    generateMetricInsight(wowEntry, metricActions, metricComments) {
      // Generate a concise insight (max ~50 words) from actions and comments for this metric
      const parts = [];
      const openActions = metricActions.filter(a => a.status?.match(/open|not started|in progress/i));
      const closedActions = metricActions.filter(a => a.status?.match(/complete|closed|done/i));
      const updateCount = metricComments.length;

      // Analyze comment themes for focus areas
      const commentText = metricComments.map(c => (c.body || '') + ' ' + (c.title || '')).join(' ').toLowerCase();
      const themes = [];
      if (commentText.match(/staff|headcount|labor|attrition|hiring/)) themes.push('staffing');
      if (commentText.match(/equipment|trailer|door|machine|conveyor/)) themes.push('equipment');
      if (commentText.match(/process|rotation|standard|procedure|training/)) themes.push('process');
      if (commentText.match(/volume|peak|surge|capacity/)) themes.push('volume/capacity');
      if (commentText.match(/quality|error|defect|damage/)) themes.push('quality');
      if (commentText.match(/dock|inbound|outbound|receive|ship/)) themes.push('dock ops');

      // Build insight
      if (metricActions.length === 0 && wowEntry.status === 'critical') {
        parts.push('No actions defined despite critical status.');
        parts.push('Needs immediate action plan creation with clear owners.');
      } else if (metricActions.length === 0 && (wowEntry.status === 'warning')) {
        parts.push('No actions defined.');
        parts.push('Create countermeasures to address gap.');
      } else if (openActions.length > 0 && wowEntry.change < 0) {
        parts.push(`${openActions.length} open action(s) not yet driving improvement.`);
        if (themes.length > 0) parts.push(`Focus: ${themes.slice(0, 2).join(', ')}.`);
      } else if (openActions.length > 0 && wowEntry.change > 0) {
        parts.push(`${openActions.length} open action(s) — performance improving.`);
        if (themes.length > 0) parts.push(`Key themes: ${themes.slice(0, 2).join(', ')}.`);
      } else if (closedActions.length > 0 && openActions.length === 0 && wowEntry.status !== 'on-track' && wowEntry.status !== 'exceeding') {
        parts.push(`All ${closedActions.length} actions closed but still behind.`);
        parts.push('New countermeasures needed.');
      } else if (wowEntry.status === 'on-track' || wowEntry.status === 'exceeding') {
        parts.push('On track.');
        if (closedActions.length > 0) parts.push(`${closedActions.length} action(s) completed successfully.`);
        if (themes.length > 0) parts.push(`Sustained by: ${themes.slice(0, 2).join(', ')}.`);
      }

      // Add update count context
      if (updateCount === 0) {
        parts.push('No weekly updates — needs owner engagement.');
      } else if (updateCount < 4) {
        parts.push(`Only ${updateCount} update(s) — below expected cadence.`);
      }

      // Focus recommendation based on themes and status
      if (parts.length === 0) {
        if (themes.length > 0) {
          parts.push(`Focus area: ${themes.slice(0, 2).join(', ')}.`);
        } else {
          parts.push('No clear signal from updates.');
        }
      }

      // Trim to approximately 50 words
      let insight = parts.join(' ');
      const words = insight.split(/\s+/);
      if (words.length > 50) {
        insight = words.slice(0, 50).join(' ') + '…';
      }

      return insight;
    }

    copyWeeklySummaryToClipboard(summary, siteLabel) {
      const { total, behind, ahead, highlights, lowlights, improving, declining, wowChanges } = summary;
      const dateStr = new Date().toLocaleDateString('en-US', { weekday: 'short', month: 'short', day: 'numeric', year: 'numeric' });
      const actions = summary.actions || [];
      const comments = this._cachedComments || [];

      const btn = document.getElementById('rtb-copy-summary-btn');
      const origText = btn.textContent;
      btn.textContent = '⏳ Rendering...';
      btn.disabled = true;

      // Build all metrics sorted: behind first, then rest, then ahead
      const highlightNames = new Set(highlights.map(h => h.name));
      const lowlightNames = new Set(lowlights.map(l => l.name));
      const remaining = wowChanges.filter(w => !highlightNames.has(w.name) && !lowlightNames.has(w.name));
      remaining.sort((a, b) => (a.varianceFromTarget ?? 0) - (b.varianceFromTarget ?? 0));
      const allMetrics = [...lowlights, ...remaining, ...highlights];

      // --- Build a 768px-wide dark-themed visual (matches the popup) ---
      const container = document.createElement('div');
      container.style.cssText = 'position:fixed;left:-9999px;top:0;z-index:-1;width:768px;font-family:-apple-system,BlinkMacSystemFont,"Segoe UI",Roboto,sans-serif;';
      container.innerHTML = this.buildPrintVisual(siteLabel, dateStr, total, ahead, behind, improving, declining, allMetrics, actions, comments);
      document.body.appendChild(container);

      // Give browser a tick to layout, then capture
      setTimeout(async () => {
        try {
          const canvas = await html2canvas(container, {
            backgroundColor: '#1a1a2e',
            scale: 2,
            useCORS: true,
            logging: false,
            width: 768,
          });

          document.body.removeChild(container);

          canvas.toBlob(async (blob) => {
            if (blob && navigator.clipboard && navigator.clipboard.write) {
              try {
                await navigator.clipboard.write([
                  new ClipboardItem({ 'image/png': blob })
                ]);
                btn.textContent = '✅ Copied! Ctrl+V in Word';
                btn.disabled = false;
                setTimeout(() => { btn.textContent = origText; }, 3000);
              } catch (clipErr) {
                this.openScreenshotInTab(canvas);
                btn.textContent = '✅ Opened in tab';
                btn.disabled = false;
                setTimeout(() => { btn.textContent = origText; }, 3000);
              }
            } else {
              this.openScreenshotInTab(canvas);
              btn.textContent = '✅ Opened in tab';
              btn.disabled = false;
              setTimeout(() => { btn.textContent = origText; }, 3000);
            }
          }, 'image/png');
        } catch (err) {
          console.error('[RTB Analyzer] Render error:', err);
          document.body.removeChild(container);
          btn.textContent = '❌ Failed';
          btn.disabled = false;
          setTimeout(() => { btn.textContent = origText; }, 3000);
        }
      }, 100);
    }

    async fetchAllActions() {
      const allActions = [];
      const metrics = this.analysisResults;

      if (metrics.length === 0) return allActions;

      console.log('[RTB Analyzer] fetchAllActions starting with', metrics.length, 'metrics');
      console.log('[RTB Analyzer] First metric:', JSON.stringify({ name: metrics[0].name, actionPlanId: metrics[0].actionPlanId, hasElement: !!metrics[0].element }));

      // Debug: count all action-plan links on page
      const apLinks = document.querySelectorAll('a[href*="action-plan"]');

      const progressEl = document.getElementById('rtb-actions-progress');

      for (let i = 0; i < metrics.length; i++) {

        if (progressEl) {
          progressEl.textContent = `Fetching actions for ${metric.name} (${i + 1}/${metrics.length})...`;
        }

        // Find the link for this metric's action plan
        const link = this.findMetricLink(metric);
        if (!link) {
          console.log('[RTB Analyzer] No link found for metric:', metric.name, '| actionPlanId:', metric.actionPlanId);
          continue;
        }

        console.log('[RTB Analyzer] Found link for', metric.name, ':', link);

        try {
          const actions = await this.fetchActionsViaIframe(link, metric.name);
          actions.forEach(a => { a.metric = metric.name; });
          allActions.push(...actions);
          console.log('[RTB Analyzer] Got', actions.length, 'actions for', metric.name);
        } catch (err) {
          console.warn('[RTB Analyzer] Failed to fetch actions for', metric.name, err.message);
        }

        // Delay between fetches
        if (i < metrics.length - 1) {
          await new Promise(r => setTimeout(r, 500));
        }
      }

      console.log('[RTB Analyzer] fetchAllActions complete:', allActions.length, 'total actions');
      if (progressEl) progressEl.textContent = 'Done.';
      this._cachedActions = allActions;
      return allActions;
    }

    findMetricLink(metric) {
      // If we have an action plan ID, construct the URL directly
      if (metric.actionPlanId && metric.actionPlanId.length >= 10) {
        return `https://rtb.ars-pe.amazon.dev/action-plan-details-v3/${metric.actionPlanId}`;
      }

      // Look for a link containing the action plan ID
      const allLinks = document.querySelectorAll('a[href]');

      if (metric.actionPlanId) {
        for (const link of allLinks) {
          const text = link.textContent.trim();
          if (text === metric.actionPlanId || text.includes(metric.actionPlanId)) {
            return link.href;
          }
        }

        // Try finding by href containing the action plan ID
        for (const link of allLinks) {
          if (link.href && link.href.includes(metric.actionPlanId)) {
            return link.href;
          }
        }
      }

      // Try finding by the metric element's row
      if (metric.element) {
        const link = metric.element.querySelector('a[href*="action-plan"]');
        if (link) return link.href;
        // Also try any link in the row
        const anyLink = metric.element.querySelector('a[href]');
        if (anyLink && anyLink.href.includes('action-plan')) return anyLink.href;
      }

      // Try finding link by metric name text in links
      if (metric.name) {
        for (const link of allLinks) {
          const text = link.textContent.trim();
          if (text === metric.name && link.href.includes('action-plan')) {
            return link.href;
          }
        }
        // Try partial match on metric name
        for (const link of allLinks) {
          const text = link.textContent.trim();
          if (metric.name.length > 5 && text.includes(metric.name) && link.href.includes('action-plan')) {
            return link.href;
          }
        }
      }

      // Walk up from metric name text element to find nearby action-plan links
      if (metric.name) {
        const walker = document.createTreeWalker(document.body, NodeFilter.SHOW_TEXT, null);
        while (walker.nextNode()) {
          if (walker.currentNode.textContent.trim() === metric.name) {
            let parent = walker.currentNode.parentElement;
            for (let i = 0; i < 6 && parent; i++) {
              const link = parent.querySelector('a[href*="action-plan"]');
              if (link) return link.href;
              parent = parent.parentElement;
            }
            break;
          }
        }
      }

      // Last resort: if metric element exists, walk up to find any row-level container with a link
      if (metric.element) {
        let el = metric.element;
        for (let i = 0; i < 3 && el; i++) {
          const links = el.querySelectorAll('a[href]');
          for (const link of links) {
            if (link.href.includes('action-plan') || link.href.includes('/v3/')) {
              return link.href;
            }
          }
          el = el.parentElement;
        }
      }

      console.log('[RTB Analyzer] findMetricLink FAILED for:', metric.name, '| actionPlanId:', metric.actionPlanId, '| hasElement:', !!metric.element);
      return null;
    }

    fetchActionsViaIframe(url, metricName) {
      // Fetch action plan details via the RTB API (POST to /api/v3/actionplan/get)
      return new Promise(async (resolve) => {
        const idMatch = url.match(/([a-f0-9]{20,})/i);
        const actionPlanId = idMatch ? idMatch[1] : null;
        if (!actionPlanId) {
          resolve([]);
          return;
        }

        try {
          const body = JSON.stringify({ actionPlanId: actionPlanId });
          const encoder = new TextEncoder();
          const data = encoder.encode(body);
          const hashBuffer = await crypto.subtle.digest('SHA-256', data);
          const hashArray = Array.from(new Uint8Array(hashBuffer));
          const hashHex = hashArray.map(b => b.toString(16).padStart(2, '0')).join('');

          const response = await fetch('https://rtb.ars-pe.amazon.dev/api/v3/actionplan/get', {
            method: 'POST',
            headers: {
              'Content-Type': 'application/json',
              'x-amz-content-sha256': hashHex,
            },
            credentials: 'include',
            body,
          });

          if (response.ok) {
            const result = await response.json();
            const actions = (result.actions || []).map(a => ({
              name: a.actionName || a.name || '',
              description: a.description || '',
              status: a.status || '',
              assignee: a.assignee || '',
              priority: a.priority || '',
              dueDate: a.dueDate || '',
              speAudit: a.speAudit || '',
              userImpact: a.userImpact || '',
            }));
            const comments = (result.comments || []).map(c => ({
              author: c.createdBy || '',
              timestamp: c.createdAt ? new Date(c.createdAt * 1000).toLocaleString() : '',
              title: c.title || '',
              body: c.body || '',
              tags: [c.role, c.type].filter(Boolean),
              metric: metricName,
            }));

            if (comments.length > 0) {
              if (!this._cachedComments) this._cachedComments = [];
              this._cachedComments.push(...comments);
            }

            console.log('[RTB Analyzer] Got', actions.length, 'actions and', comments.length, 'comments for', metricName);
            resolve(actions);
          } else {
            console.log('[RTB Analyzer] API error for', metricName, '- status:', response.status);
            resolve([]);
          }
        } catch (err) {
          console.log('[RTB Analyzer] Fetch error for', metricName, ':', err.message);
          resolve([]);
        }
      });
    }

    parseActionsFromAPI(data, metricName) {
      const actions = [];

      // Try multiple possible structures for actions in the response
      const possibleArrays = [
        data.actions,
        data.actionItems,
        data.actionPlan?.actions,
        data.data?.actions,
        data.result?.actions,
        data.items,
      ];

      // Also check if the response itself is an array
      if (Array.isArray(data)) {
        possibleArrays.unshift(data);
      }

      // Check all nested objects for arrays that look like actions
      for (const [key, val] of Object.entries(data)) {
        if (Array.isArray(val) && val.length > 0 && typeof val[0] === 'object') {
          const sample = val[0];
          if (sample.actionName || sample.name || sample.title || sample.description || sample.status) {
            possibleArrays.push(val);
          }
        }
      }

      for (const arr of possibleArrays) {
        if (Array.isArray(arr) && arr.length > 0) {
          console.log('[RTB Analyzer] Found action array for', metricName, '- length:', arr.length, '| sample keys:', Object.keys(arr[0]).join(', '));
          arr.forEach(a => {
            if (typeof a !== 'object' || a === null) return;
            actions.push({
              name: a.actionName || a.name || a.title || a.action || '',
              description: a.description || a.details || a.notes || '',
              status: a.status || a.actionStatus || '',
              assignee: a.assignee || a.owner || a.assignedTo || '',
              priority: a.priority || '',
              dueDate: a.dueDate || a.targetDate || a.due_date || '',
              speAudit: a.speAudit || a.spe_audit || '',
              userImpact: a.userImpact || a.impact || a.user_impact || '',
            });
          });
          if (actions.length > 0) break;
        }
      }

      console.log('[RTB Analyzer] parseActionsFromAPI for', metricName, ':', actions.length, 'actions');
      return actions;
    }

    parseActionsFromHTML(html, metricName) {
      const actions = [];
      // Parse the HTML string into a DOM
      const parser = new DOMParser();
      const doc = parser.parseFromString(html, 'text/html');

      // Try to find embedded JSON data (React SSR or __NEXT_DATA__)
      const scripts = doc.querySelectorAll('script');
      for (const script of scripts) {
        const text = script.textContent;
        if (text.includes('actions') || text.includes('actionItems') || text.includes('actionPlan')) {
          try {
            // Try to find JSON objects in script content
            const jsonMatch = text.match(/\{[^{}]*"action[^{}]*\}/g);
            if (jsonMatch) {
              for (const j of jsonMatch) {
                try {
                  const obj = JSON.parse(j);
                  if (obj.actionName || obj.name) {
                    actions.push({
                      name: obj.actionName || obj.name || '',
                      description: obj.description || '',
                      status: obj.status || '',
                      assignee: obj.assignee || obj.owner || '',
                      priority: obj.priority || '',
                      dueDate: obj.dueDate || '',
                    });
                  }
                } catch (e) { /* skip */ }
              }
            }
          } catch (e) { /* skip */ }
        }
      }

      // Also try standard table extraction from HTML
      if (actions.length === 0) {
        const extracted = this.extractActionsFromDoc(doc);
        if (extracted.length > 0) return extracted;
      }

      console.log('[RTB Analyzer] parseActionsFromHTML for', metricName, ':', actions.length, 'actions');
      return actions;
    }

    extractActionsFromDoc(doc) {
      const actions = [];

      // Strategy 1: Find standard <table> with action columns
      const tables = doc.querySelectorAll('table');
      for (const table of tables) {
        const headerRow = table.querySelector('thead tr, tr:first-child');
        if (!headerRow) continue;

        const headers = Array.from(headerRow.querySelectorAll('th, td'))
          .map(h => h.textContent.trim().toLowerCase());

        const hasActionName = headers.some(h => h.includes('action name') || h === 'action' || h.includes('title'));
        if (!hasActionName) continue;

        const colMap = {
          name: headers.findIndex(h => h.includes('action name') || h === 'action' || h.includes('title') || h === 'name'),
          description: headers.findIndex(h => h.includes('description')),
          status: headers.findIndex(h => h === 'status'),
          assignee: headers.findIndex(h => h.includes('assignee') || h.includes('owner')),
          priority: headers.findIndex(h => h.includes('priority')),
          dueDate: headers.findIndex(h => h.includes('due') || (h.includes('date') && !h.includes('create'))),
          speAudit: headers.findIndex(h => h.includes('spe') || h.includes('audit')),
          userImpact: headers.findIndex(h => h.includes('impact')),
        };

        const rows = table.querySelectorAll('tbody tr, tr:not(:first-child)');
        rows.forEach(row => {
          const cells = Array.from(row.querySelectorAll('td')).map(c => c.textContent.trim());
          if (cells.length < 2) return;
          actions.push({
            name: colMap.name >= 0 ? cells[colMap.name] : '',
            description: colMap.description >= 0 ? cells[colMap.description] : '',
            status: colMap.status >= 0 ? cells[colMap.status] : '',
            assignee: colMap.assignee >= 0 ? cells[colMap.assignee] : '',
            priority: colMap.priority >= 0 ? cells[colMap.priority] : '',
            dueDate: colMap.dueDate >= 0 ? cells[colMap.dueDate] : '',
            speAudit: colMap.speAudit >= 0 ? cells[colMap.speAudit] : '',
            userImpact: colMap.userImpact >= 0 ? cells[colMap.userImpact] : '',
          });
        });

        if (actions.length > 0) return actions;
      }

      // Strategy 2: AWSUI grid-based table
      const allRows = doc.querySelectorAll('[role="row"]');
      let headerRow = null;

      for (const row of allRows) {
        const headerCells = row.querySelectorAll('[role="columnheader"]');
        if (headerCells.length > 0) {
          const headerText = row.textContent.toLowerCase();
          if (headerText.includes('action') && headerText.includes('status')) {
            headerRow = row;
            break;
          }
        }
      }

      if (headerRow) {
        const headers = Array.from(headerRow.querySelectorAll('[role="columnheader"]'))
          .map(h => h.textContent.trim().toLowerCase());

        const colMap = {
          name: headers.findIndex(h => h.includes('action name') || h === 'action' || h.includes('title')),
          description: headers.findIndex(h => h.includes('description')),
          status: headers.findIndex(h => h === 'status'),
          assignee: headers.findIndex(h => h.includes('assignee') || h.includes('owner')),
          priority: headers.findIndex(h => h.includes('priority')),
          dueDate: headers.findIndex(h => h.includes('due') || (h.includes('date') && !h.includes('create'))),
          speAudit: headers.findIndex(h => h.includes('spe') || h.includes('audit')),
          userImpact: headers.findIndex(h => h.includes('impact')),
        };

        // Get data rows that come after the header
        let foundHeader = false;
        for (const row of allRows) {
          if (row === headerRow) { foundHeader = true; continue; }
          if (!foundHeader) continue;
          // Stop if we hit another header row (different section)
          if (row.querySelector('[role="columnheader"]')) break;

          const cells = Array.from(row.querySelectorAll('[role="gridcell"], [role="cell"], td'))
            .map(c => c.textContent.trim());
          if (cells.length < 2) continue;

          actions.push({
            name: colMap.name >= 0 ? cells[colMap.name] : '',
            description: colMap.description >= 0 ? cells[colMap.description] : '',
            status: colMap.status >= 0 ? cells[colMap.status] : '',
            assignee: colMap.assignee >= 0 ? cells[colMap.assignee] : '',
            priority: colMap.priority >= 0 ? cells[colMap.priority] : '',
            dueDate: colMap.dueDate >= 0 ? cells[colMap.dueDate] : '',
            speAudit: colMap.speAudit >= 0 ? cells[colMap.speAudit] : '',
            userImpact: colMap.userImpact >= 0 ? cells[colMap.userImpact] : '',
          });
        }
      }

      return actions;
    }

    async fetchAllComments() {
      const allComments = [];
      const metrics = this.analysisResults;
      if (metrics.length === 0) return allComments;

      const progressEl = document.getElementById('rtb-comments-progress');

      for (let i = 0; i < metrics.length; i++) {
        const metric = metrics[i];
        if (progressEl) {
          progressEl.textContent = `Fetching comments for ${metric.name} (${i + 1}/${metrics.length})...`;
        }

        const link = this.findMetricLink(metric);
        if (!link) continue;

        try {
          const comments = await this.fetchCommentsViaIframe(link, metric.name);
          comments.forEach(c => { c.metric = metric.name; });
          allComments.push(...comments);
          console.log('[RTB Analyzer] Got', comments.length, 'comments for', metric.name);
        } catch (err) {
          console.warn('[RTB Analyzer] Failed to fetch comments for', metric.name, err.message);
        }

        if (i < metrics.length - 1) {
          await new Promise(r => setTimeout(r, 500));
        }
      }

      return allComments;
    }

    fetchCommentsViaIframe(url, metricName) {
      return new Promise((resolve) => {
        const iframe = document.createElement('iframe');
        iframe.style.cssText = 'position:fixed;top:-9999px;left:-9999px;width:1200px;height:800px;opacity:0;pointer-events:none;';
        iframe.src = url;

        const timeout = setTimeout(() => { cleanup(); resolve([]); }, 20000);

        const cleanup = () => {
          clearTimeout(timeout);
          clearInterval(pollInterval);
          if (iframe.parentElement) iframe.parentElement.removeChild(iframe);
        };

        let attempts = 0;
        let pollInterval;

        iframe.onload = () => {
          pollInterval = setInterval(() => {
            attempts++;
            try {
              const doc = iframe.contentDocument || iframe.contentWindow?.document;
              if (!doc || !doc.body) return;

              const comments = this.extractCommentsFromDoc(doc);
              if (comments.length > 0) {
                cleanup();
                resolve(comments);
                return;
              }

              if (attempts > 50) { cleanup(); resolve([]); }
            } catch (e) {
              if (attempts > 30) { cleanup(); resolve([]); }
            }
          }, 200);
        };

        iframe.onerror = () => { cleanup(); resolve([]); };
        document.body.appendChild(iframe);
      });
    }

    extractCommentsFromDoc(doc) {
      const comments = [];

      // Strategy: Find the "Comments" section by looking for heading text,
      // then extract individual comments by their content patterns.
      // Each comment has: tags (shift_owner, weekly), timestamp, title (bold), body, "By: username"

      // Step 1: Find all text nodes that match "By: <username>" pattern - these mark comment ends
      const allElements = doc.querySelectorAll('*');
      const byAuthorElements = [];

      for (const el of allElements) {
        const text = el.textContent.trim();
        // Look for elements whose direct text content matches "By: username"
        if (el.children.length === 0 && text.match(/^By:\s*\w+$/)) {
          byAuthorElements.push(el);
        }
      }

      console.log('[RTB Analyzer] Found', byAuthorElements.length, '"By:" elements');

      // Step 2: For each "By:" element, walk up to find the comment card container
      for (const authorEl of byAuthorElements) {
        // Walk up to find a container that holds the full comment
        let container = authorEl.parentElement;
        for (let depth = 0; depth < 6 && container; depth++) {
          const text = container.textContent.trim();
          // A comment container should have: date pattern + "By:" + substantial text
          if (text.length > 50 && text.match(/\d{1,2}\/\d{1,2}\/\d{4}/) && text.includes('By:')) {
            // Check it's not too large (not the whole page)
            if (text.length < 3000) break;
          }
          container = container.parentElement;
        }

        if (!container || container.textContent.length > 3000) continue;

        const fullText = container.textContent.trim();

        // Extract author
        const authorMatch = fullText.match(/By:\s*(\w+)/);
        const author = authorMatch ? authorMatch[1] : '';

        // Extract timestamp
        const dateMatch = fullText.match(/(\d{1,2}\/\d{1,2}\/\d{4},?\s*\d{1,2}:\d{2}:\d{2}\s*[AP]M)/i);
        const timestamp = dateMatch ? dateMatch[1] : '';

        // Extract tags - look for short badge-like text elements
        const tags = [];
        const smallEls = container.querySelectorAll('span, div');
        for (const el of smallEls) {
          const t = el.textContent.trim();
          if (t.length > 0 && t.length < 20 && el.children.length === 0) {
            if (t.match(/^(shift_owner|weekly|daily|site_leader|manager|gm|action_owner|process_engineer)$/i)) {
              tags.push(t);
            }
          }
        }

        // Extract title - look for bold/strong elements
        let title = '';
        const boldEls = container.querySelectorAll('strong, b, [class*="bold"], [class*="header"]');
        for (const el of boldEls) {
          const t = el.textContent.trim();
          if (t.length > 2 && t.length < 100 && !t.match(/^(By:|Comments)/)) {
            title = t;
            break;
          }
        }

        // Extract body - the longest text block in the container
        let body = '';
        const textBlocks = container.querySelectorAll('div, p, span');
        for (const el of textBlocks) {
          if (el.children.length > 2) continue; // Skip containers
          const t = el.textContent.trim();
          if (t.length > body.length && t.length > 30 && !t.match(/^By:/) && !t.match(/^\d{1,2}\/\d{1,2}/)) {
            body = t;
          }
        }

        // If no body found from sub-elements, use the full text minus metadata
        if (!body || body.length < 20) {
          body = fullText;
          if (author) body = body.replace(`By: ${author}`, '');
          if (timestamp) body = body.replace(timestamp, '');
          tags.forEach(t => { body = body.replace(new RegExp(t, 'g'), ''); });
          body = body.replace(/\s+/g, ' ').trim();
        }

        if (body.length < 15) continue;

        comments.push({ author, timestamp, tags, title, body, fullText });
      }

      // Deduplicate
      const seen = new Set();
      return comments.filter(c => {
        const key = c.body.substring(0, 60);
        if (seen.has(key)) return false;
        seen.add(key);
        return true;
      });
    }

    buildMetricCommentSummary(metric, comments) {
      // Build a qualitative summary: root causes, what's being tried, what's blocking, what's working
      const allBodies = comments.map(c => (c.body || c.title || '').trim()).filter(t => t.length > 10);

      if (allBodies.length === 0) return '<span style="color:#f39c12;">No substantive updates this week.</span>';

      const allText = allBodies.join(' ');
      const parts = [];

      // What's being done / tried (actions in progress)
      const doingSentences = [];
      allBodies.forEach(body => {
        const sentences = body.split(/(?<=[.!?])\s+/).filter(s => s.length > 15);
        sentences.forEach(s => {
          if (s.match(/\b(implement|focus|started|launched|doing|working on|rolling out|executing|enforcing|tracking|monitoring|will|plan to|targeting|assigned|added|created|established|ensure|maintain)\b/i)) {
            doingSentences.push(s);
          }
        });
      });

      // What's working / results
      const resultSentences = [];
      allBodies.forEach(body => {
        const sentences = body.split(/(?<=[.!?])\s+/).filter(s => s.length > 15);
        sentences.forEach(s => {
          if (s.match(/\b(improved|increased|resulted|achieved|moved|performing at|rate of|seeing|saw|gained|better|up from|above|ahead|exceed|sustain|consistent)\b/i)) {
            resultSentences.push(s);
          }
        });
      });

      // What's blocking / root causes
      const blockerSentences = [];
      allBodies.forEach(body => {
        const sentences = body.split(/(?<=[.!?])\s+/).filter(s => s.length > 15);
        sentences.forEach(s => {
          if (s.match(/\b(challenge|barrier|issue|problem|due to|caused by|lack|shortage|unable|difficult|impacted by|driven by|because|reason|root cause|attrition|volume|capacity|equipment|staffing|short|miss)\b/i)) {
            blockerSentences.push(s);
          }
        });
      });

      // Deduplicate and limit each bucket
      const dedup = (arr, max) => {
        const seen = new Set();
        return arr.filter(s => {
          const key = s.substring(0, 40).toLowerCase();
          if (seen.has(key)) return false;
          seen.add(key);
          return true;
        }).slice(0, max);
      };

      const doing = dedup(doingSentences, 3);
      const results = dedup(resultSentences, 2);
      const blockers = dedup(blockerSentences, 2);

      // Build the summary paragraph
      if (blockers.length > 0) {
        parts.push(`<strong>Challenges:</strong> ${blockers.join(' ')}`);
      }
      if (doing.length > 0) {
        parts.push(`<strong>What's being done:</strong> ${doing.join(' ')}`);
      }
      if (results.length > 0) {
        parts.push(`<strong>Results:</strong> ${results.join(' ')}`);
      }

      // Shift callout only if tied to a problem/win
      const shiftPattern = /\b(FHD|BHD|FHN|BHN|day\s*shift|night\s*shift|front half|back half)\b/gi;
      const shiftMentions = allText.match(shiftPattern);
      if (shiftMentions) {
        const shiftComments = comments.filter(c => ((c.body || '') + ' ' + (c.title || '')).match(shiftPattern));
        const shiftIssues = shiftComments.filter(c => (c.body || '').match(/behind|issue|problem|drop|miss|struggle|lower|poor|challenge/i));
        const shiftWins = shiftComments.filter(c => (c.body || '').match(/improve|better|ahead|strong|exceed|above|gain|success/i));
        const uniqueShifts = [...new Set(shiftMentions.map(s => s.toUpperCase()))];

        if (shiftIssues.length > 0) {
          parts.push(`<span style="color:#e74c3c;"><strong>Shift:</strong> ${uniqueShifts.join(', ')} identified as source of gap.</span>`);
        } else if (shiftWins.length > 0) {
          parts.push(`<span style="color:#27ae60;"><strong>Shift:</strong> ${uniqueShifts.join(', ')} driving improvement.</span>`);
        }
      }

      // If we got nothing from pattern matching, just grab the most substantive sentences overall
      if (parts.length === 0) {
        const allSentences = [];
        allBodies.forEach(body => {
          body.split(/(?<=[.!?])\s+/).filter(s => s.length > 25).forEach(s => allSentences.push(s));
        });
        const best = dedup(allSentences, 4);
        return best.join(' ') || '<span style="color:#888;">Comments present but no actionable detail extracted.</span>';
      }

      return parts.join('<br style="margin-bottom:6px;">');
    }

    // ---- Weekly Author Tracker ----
    buildWeeklyAuthorTracker(comments) {
      // Amazon fiscal weeks run Sunday–Saturday.
      // Determine the past 4 fiscal weeks ending before or including the current week.
      const getFiscalWeekStart = (date) => {
        const d = new Date(date);
        const day = d.getDay(); // 0=Sun
        d.setDate(d.getDate() - day);
        d.setHours(0, 0, 0, 0);
        return d;
      };

      const now = new Date();
      const currentWeekStart = getFiscalWeekStart(now);

      // Build 4 weeks: current week and 3 prior
      const weeks = [];
      for (let w = 0; w < 4; w++) {
        const start = new Date(currentWeekStart);
        start.setDate(start.getDate() - (w * 7));
        const end = new Date(start);
        end.setDate(end.getDate() + 6);
        end.setHours(23, 59, 59, 999);
        weeks.push({ start, end, label: `WK ${this.formatShortDate(start)}` });
      }
      weeks.reverse(); // oldest first

      // Map comments into weeks by author
      // weekAuthors[weekIdx] = { authorName: count }
      const weekAuthors = weeks.map(() => ({}));
      const allAuthors = new Set();

      comments.forEach(c => {
        if (!c.author || !c.timestamp) return;
        const d = new Date(c.timestamp);
        if (isNaN(d.getTime())) return;
        const author = c.author.trim();
        if (!author) return;
        allAuthors.add(author);

        for (let i = 0; i < weeks.length; i++) {
          if (d >= weeks[i].start && d <= weeks[i].end) {
            weekAuthors[i][author] = (weekAuthors[i][author] || 0) + 1;
            break;
          }
        }
      });

      const sortedAuthors = [...allAuthors].sort((a, b) => a.toLowerCase().localeCompare(b.toLowerCase()));

      if (sortedAuthors.length === 0) {
        return `<div class="rtb-comment-block" style="border-left:4px solid #f39c12;margin-bottom:20px;">
          <h4 style="color:#f39c12;">📋 Weekly Comment Tracker</h4>
          <p style="color:#aaa;font-size:13px;">No author/timestamp data available to build weekly tracker.</p>
        </div>`;
      }

      // Build the table
      let tbl = `<div class="rtb-comment-block" style="border-left:4px solid #3498db;margin-bottom:24px;">
        <h4 style="color:#3498db;margin-bottom:12px;">📋 Weekly Comment Tracker — Who's Updating?</h4>
        <p style="color:#aaa;font-size:12px;margin-bottom:10px;">Tracks which ops managers added comments each fiscal week (Sun–Sat). ✅ = commented, ❌ = no comment found.</p>
        <table class="rtb-table" style="font-size:13px;">
          <thead><tr>
            <th style="min-width:120px;">Author</th>`;

      weeks.forEach(w => {
        tbl += `<th style="text-align:center;">${w.label}</th>`;
      });
      tbl += `<th style="text-align:center;">Weeks Active</th></tr></thead><tbody>`;

      sortedAuthors.forEach(author => {
        let activeCount = 0;
        tbl += `<tr><td style="font-weight:bold;color:#eee;">${author}</td>`;
        weeks.forEach((_, i) => {
          const count = weekAuthors[i][author] || 0;
          if (count > 0) {
            activeCount++;
            tbl += `<td style="text-align:center;"><span style="color:#2ecc71;font-size:16px;">✅</span><span style="font-size:11px;color:#aaa;margin-left:4px;">(${count})</span></td>`;
          } else {
            tbl += `<td style="text-align:center;"><span style="color:#e74c3c;font-size:16px;">❌</span></td>`;
          }
        });

        // Weeks active score with color coding
        const pct = (activeCount / weeks.length) * 100;
        const scoreColor = pct === 100 ? '#2ecc71' : pct >= 50 ? '#f39c12' : '#e74c3c';
        tbl += `<td style="text-align:center;color:${scoreColor};font-weight:bold;">${activeCount}/${weeks.length}</td>`;
        tbl += `</tr>`;
      });

      tbl += `</tbody></table></div>`;
      return tbl;
    }

    }

    async fetchAllTrends() {
      const allTrends = [];
      const metrics = this.analysisResults;

      if (metrics.length === 0) return allTrends;

      const progressEl = document.getElementById('rtb-trends-progress');

      for (let i = 0; i < metrics.length; i++) {
        const metric = metrics[i];
        if (progressEl) {
          progressEl.textContent = `Fetching trend data for ${metric.name} (${i + 1}/${metrics.length})...`;
        }

        const link = this.findMetricLink(metric);
        if (!link) continue;

        try {
          const trendData = await this.fetchTrendViaIframe(link, metric.name);
          if (trendData && trendData.length > 0) {
            allTrends.push({ metric: metric.name, path: metric.path, data: trendData });
            console.log('[RTB Analyzer] Got', trendData.length, 'weeks of trend data for', metric.name);
          }
        } catch (err) {
          console.warn('[RTB Analyzer] Failed to fetch trends for', metric.name, err.message);
        }

        if (i < metrics.length - 1) {
          await new Promise(r => setTimeout(r, 500));
        }
      }

      return allTrends;
    }

    fetchTrendViaIframe(url, metricName) {
      return new Promise((resolve) => {
        const iframe = document.createElement('iframe');
        iframe.style.cssText = 'position:fixed;top:-9999px;left:-9999px;width:1200px;height:800px;opacity:0;pointer-events:none;';
        iframe.src = url;

        const timeout = setTimeout(() => {
          cleanup();
          resolve([]);
        }, 20000);

        const cleanup = () => {
          clearTimeout(timeout);
          clearInterval(pollInterval);
          if (iframe.parentElement) iframe.parentElement.removeChild(iframe);
        };

        let attempts = 0;
        let pollInterval;

        iframe.onload = () => {
          // First, try to click "Weekly Data" button in the iframe
          setTimeout(() => {
            try {
              const doc = iframe.contentDocument || iframe.contentWindow?.document;
              if (!doc) return;
              // Look for "Weekly Data" button and click it
              const buttons = doc.querySelectorAll('button, [role="button"], [class*="button"]');
              for (const btn of buttons) {
                if (btn.textContent.trim().match(/weekly data/i)) {
                  btn.click();
                  console.log('[RTB Analyzer] Clicked Weekly Data button for', metricName);
                  break;
                }
              }
            } catch (e) { /* ignore */ }
          }, 2000);

          // Start polling for the weekly data table
          pollInterval = setInterval(() => {
            attempts++;
            try {
              const doc = iframe.contentDocument || iframe.contentWindow?.document;
              if (!doc || !doc.body) return;

              const trendData = this.extractWeeklyDataFromDoc(doc);
              if (trendData.length > 0) {
                cleanup();
                resolve(trendData);
                return;
              }

              if (attempts > 60) { // 12 seconds after load
                cleanup();
                resolve([]);
              }
            } catch (e) {
              if (attempts > 30) {
                cleanup();
                resolve([]);
              }
            }
          }, 200);
        };

        iframe.onerror = () => { cleanup(); resolve([]); };
        document.body.appendChild(iframe);
      });
    }

    extractWeeklyDataFromDoc(doc) {
      const data = [];

      // Look for a table with columns: Date, Weekly Performance, Glidepath, Target, Variance from Target
      const tables = doc.querySelectorAll('table');
      for (const table of tables) {
        const headerRow = table.querySelector('thead tr, tr:first-child');
        if (!headerRow) continue;

        const headers = Array.from(headerRow.querySelectorAll('th, td'))
          .map(h => h.textContent.trim().toLowerCase());

        const hasDate = headers.some(h => h.includes('date'));
        const hasPerformance = headers.some(h => h.includes('performance') || h.includes('weekly'));
        if (!hasDate || !hasPerformance) continue;

        const colMap = {
          date: headers.findIndex(h => h.includes('date')),
          performance: headers.findIndex(h => h.includes('performance') || h.includes('weekly')),
          glidepath: headers.findIndex(h => h.includes('glidepath') || h.includes('glide')),
          target: headers.findIndex(h => h === 'target'),
          variance: headers.findIndex(h => h.includes('variance')),
        };

        const rows = table.querySelectorAll('tbody tr, tr:not(:first-child)');
        rows.forEach(row => {
          const cells = Array.from(row.querySelectorAll('td')).map(c => c.textContent.trim());
          if (cells.length < 2) return;

          const date = colMap.date >= 0 ? cells[colMap.date] : '';
          const performance = colMap.performance >= 0 ? parseFloat(cells[colMap.performance]) : null;
          const glidepath = colMap.glidepath >= 0 ? parseFloat(cells[colMap.glidepath]) : null;
          const target = colMap.target >= 0 ? parseFloat(cells[colMap.target]) : null;
          const variance = colMap.variance >= 0 ? parseFloat(cells[colMap.variance]) : null;

          if (date && performance !== null && !isNaN(performance)) {
            data.push({ date, performance, glidepath, target, variance });
          }
        });

        if (data.length > 0) return data;
      }

      // Try AWSUI grid-based table
      const allRows = doc.querySelectorAll('[role="row"]');
      let headerRow = null;

      for (const row of allRows) {
        const headerCells = row.querySelectorAll('[role="columnheader"]');
        if (headerCells.length > 0) {
          const text = row.textContent.toLowerCase();
          if (text.includes('date') && (text.includes('performance') || text.includes('weekly'))) {
            headerRow = row;
            break;
          }
        }
      }

      if (headerRow) {
        const headers = Array.from(headerRow.querySelectorAll('[role="columnheader"]'))
          .map(h => h.textContent.trim().toLowerCase());

        const colMap = {
          date: headers.findIndex(h => h.includes('date')),
          performance: headers.findIndex(h => h.includes('performance') || h.includes('weekly')),
          glidepath: headers.findIndex(h => h.includes('glidepath')),
          target: headers.findIndex(h => h === 'target'),
          variance: headers.findIndex(h => h.includes('variance')),
        };

        let foundHeader = false;
        for (const row of allRows) {
          if (row === headerRow) { foundHeader = true; continue; }
          if (!foundHeader) continue;
          if (row.querySelector('[role="columnheader"]')) break;

          const cells = Array.from(row.querySelectorAll('[role="gridcell"], [role="cell"], td'))
            .map(c => c.textContent.trim());
          if (cells.length < 2) continue;

          const date = colMap.date >= 0 ? cells[colMap.date] : '';
          const performance = colMap.performance >= 0 ? parseFloat(cells[colMap.performance]) : null;
          const glidepath = colMap.glidepath >= 0 ? parseFloat(cells[colMap.glidepath]) : null;
          const target = colMap.target >= 0 ? parseFloat(cells[colMap.target]) : null;
          const variance = colMap.variance >= 0 ? parseFloat(cells[colMap.variance]) : null;

          if (date && performance !== null && !isNaN(performance)) {
            data.push({ date, performance, glidepath, target, variance });
          }
        }
      }

      return data;
    }

          else if (recent < earlier * 0.95) { trendDirection = 'declining'; trendColor = '#e74c3c'; }
        }

        const latest = performances[performances.length - 1] || 0;
        const varianceFromTarget = target ? (latest - target).toFixed(2) : '—';
        const pctOfTarget = target ? ((latest / target) * 100).toFixed(1) : '—';

        html += `<div class="rtb-comment-block" style="border-left-color:${trendColor};">

      document.getElementById('sec-trends').innerHTML = html;
    }

    renderTrendChart(data, target) {
      if (!data || data.length === 0) return '';

      const performances = data.map(d => d.performance);
      const glidepaths = data.map(d => d.glidepath).filter(v => v && !isNaN(v));
      const allValues = [...performances.filter(v => !isNaN(v))];
      if (glidepaths.length > 0) allValues.push(...glidepaths);
      if (target && !isNaN(target)) allValues.push(target);

      const max = Math.max(...allValues) || 1;
      const min = Math.min(...allValues.filter(v => v > 0)) || 0;
      // Add 10% padding to top and bottom
      const padding = (max - min) * 0.1 || 1;
      const chartMin = Math.max(0, min - padding);
      const chartMax = max + padding;
      const range = chartMax - chartMin || 1;

      const svgWidth = 800;
      const svgHeight = 200;
      const padX = 40;
      const padY = 20;
      const plotWidth = svgWidth - padX * 2;
      const plotHeight = svgHeight - padY * 2;

      const toX = (i) => padX + (i / (data.length - 1 || 1)) * plotWidth;
      const toY = (val) => {
        if (!val || isNaN(val)) return null;
        return padY + plotHeight - ((val - chartMin) / range) * plotHeight;
      };

      // Y-axis labels
      const ySteps = 4;
      let yLabels = '';
      for (let i = 0; i <= ySteps; i++) {
        const val = chartMin + (range * i / ySteps);
        const y = padY + plotHeight - (i / ySteps) * plotHeight;
        yLabels += `<text x="${padX - 5}" y="${y + 3}" fill="#666" font-size="10" text-anchor="end">${val.toFixed(0)}</text>`;
        yLabels += `<line x1="${padX}" y1="${y}" x2="${svgWidth - padX}" y2="${y}" stroke="#2a2a4a" stroke-width="0.5"/>`;
      }

      // X-axis date labels (show every few)
      let xLabels = '';
      const labelInterval = Math.max(1, Math.floor(data.length / 6));
      data.forEach((d, i) => {
        if (i % labelInterval === 0 || i === data.length - 1) {
          const x = toX(i);
          const shortDate = d.date.replace(/,?\s*\d{4}/, '').trim();
          xLabels += `<text x="${x}" y="${svgHeight - 4}" fill="#666" font-size="9" text-anchor="middle">${shortDate}</text>`;
        }
      });

      // Performance line (blue)
      let perfPath = '';
      let perfDots = '';
      data.forEach((d, i) => {
        const x = toX(i);
        const y = toY(d.performance);
        if (y === null) return;
        perfPath += (perfPath ? ' L' : 'M') + ` ${x} ${y}`;
        perfDots += `<circle cx="${x}" cy="${y}" r="2.5" fill="#3498db"/>`;
      });

      // Glidepath line (red)
      let glidePathStr = '';
      data.forEach((d, i) => {
        const x = toX(i);
        const y = toY(d.glidepath);
        if (y === null) return;
        glidePathStr += (glidePathStr ? ' L' : 'M') + ` ${x} ${y}`;
      });

      // Target line (dashed)
      const targetY = toY(target);
      const targetLine = targetY !== null
        ? `<line x1="${padX}" y1="${targetY}" x2="${svgWidth - padX}" y2="${targetY}" stroke="#888" stroke-width="1" stroke-dasharray="6,4"/>`
        : '';

      return `<div style="margin:10px 0;">
        <svg viewBox="0 0 ${svgWidth} ${svgHeight}" style="width:100%;height:180px;background:#0a0a1a;border-radius:6px;border:1px solid #2a2a4a;">
          ${yLabels}
          ${xLabels}
          ${targetLine}
          ${glidePathStr ? `<path d="${glidePathStr}" fill="none" stroke="#e74c3c" stroke-width="2" opacity="0.8"/>` : ''}
          ${perfPath ? `<path d="${perfPath}" fill="none" stroke="#3498db" stroke-width="2.5"/>` : ''}
          ${perfDots}
        </svg>
        <div style="display:flex;gap:16px;font-size:11px;color:#888;margin-top:6px;">
          <span><span style="color:#3498db;">●━</span> Performance</span>
          <span><span style="color:#e74c3c;">━</span> Glidepath</span>
          ${target ? `<span><span style="color:#888;">┈┈</span> Target (${target})</span>` : ''}
        </div>
      </div>`;
    }

    renderTrendTable(data) {

    async fetchSiteDataViaAPI(site) {
      const body = JSON.stringify({ site, status: 'open', limit: 100 });

      // Compute SHA-256 hash of the body (required by the API)
      const encoder = new TextEncoder();
      const data = encoder.encode(body);
      const hashBuffer = await crypto.subtle.digest('SHA-256', data);
      const hashArray = Array.from(new Uint8Array(hashBuffer));
      const hashHex = hashArray.map(b => b.toString(16).padStart(2, '0')).join('');

      const response = await fetch('https://rtb.ars-pe.amazon.dev/api/v3/actionplan/list', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'x-amz-content-sha256': hashHex,
        },
        credentials: 'include',
        body,
      });

      if (!response.ok) throw new Error(`API returned ${response.status}`);
      const responseData = await response.json();
      const actionPlans = responseData.actionPlans || [];

      // Log first action plan's keys to discover the ID field
      if (actionPlans.length > 0) {
        const ap0 = actionPlans[0];
        console.log('[RTB Analyzer] API response sample keys:', Object.keys(ap0).join(', '));
        console.log('[RTB Analyzer] Potential ID fields:', ap0.actionPlanId || ap0.id || ap0._id || ap0.planId || ap0.uuid || 'NONE FOUND');
      }

      // Parse API response into our metric format
      const metrics = actionPlans.map(ap => {
        const perf = ap.performance || {};
        const weeklyData = perf.performanceData?.weekly || [];
        const latestWeekly = weeklyData.length > 0 ? weeklyData[weeklyData.length - 1]?.value : null;
        const target = ap.targetInfo?.target;
        const pctToTarget = ap.targetInfo?.percentToTarget;

        // Determine status
        let status = 'unknown';
        if (latestWeekly && target) {
          if (latestWeekly >= target) status = 'on-track';
          else if (latestWeekly >= target * 0.9) status = 'warning';
          else status = 'critical';
        }

        return {

          owner: ap.owner,
          startDate: ap.startDate,
          wowTrend: perf.wowTrend,
          dodTrend: perf.dodTrend,
          pctToTarget: pctToTarget ? { direction: pctToTarget >= 0 ? 'ahead' : 'behind', percentage: Math.abs(pctToTarget) } : null,

        };
      });

      const trends = actionPlans.map(ap => {

          })),
        };
      }).filter(Boolean);

      return { site, metrics, actions: [], trends, comments: [] };
    }

  function init() {
    console.log('[RTB Analyzer] Initializing...');

    const extractor = new RTBDataExtractor();
    const ui = new RTBAnalyzerUI(extractor);

    // Intercept fetch/XHR for dynamic data updates
    const origFetch = window.fetch;
    window.fetch = async function (...args) {
      const response = await origFetch.apply(this, args);
      try {
        const url = typeof args[0] === 'string' ? args[0] : args[0]?.url || '';
        if (url.match(/metric|kpi|rtb|benchmark|performance|dashboard/i)) {
          setTimeout(() => {
            window.dispatchEvent(new CustomEvent('rtb-data-updated'));
          }, 1000);
        }
      } catch (e) { /* ignore */ }
      return response;
    };

    // Listen for dynamic updates (disabled - using API mode)
    // window.addEventListener('rtb-data-updated', () => { ... });

    // Watch for DOM changes (disabled - using API mode, DOM changes don't affect data)

    // Keyboard shortcut: Ctrl+Shift+R to toggle
    document.addEventListener('keydown', (e) => {
      if (e.ctrlKey && e.shiftKey && e.key === 'R') {
        e.preventDefault();
        const overlay = document.getElementById('rtb-analyzer-overlay');
        if (overlay.classList.contains('active')) {
          overlay.classList.remove('active');
        } else {
          overlay.classList.add('active');
          ui.refresh();
        }
      }
    });

    console.log('[RTB Analyzer] Ready. Click 📈 button or press Ctrl+Shift+R to open.');
  }

  // Wait for page to be fully loaded
  if (document.readyState === 'complete') {
    init();
  } else {
    window.addEventListener('load', () => setTimeout(init, 1000));
  }
